解法一 枚举法

只有4个数字，其全排列也就24种。

4个数字间，有3个运算符的位置。每个位置有4种运算符可填。那么有4^3=64种。

到此，共有24*64=1536种。

最后，目测，插入括号的种数也不会超过1000。

这样总可能数不超过1536000种。基本能通过。

计算插入括号的种数。

例如，要将括号插入到表达式a+b+c+d中。

注意，此例在于演示括号位置，不必在意+，可以选其它运算符，也可以是不同的运算符。

第一种，括号里有一个运算符，共4种：

只有一对括号，有3种类型：

(a+b)+c+d a+(b+c)+d a+b+(c+d)

有两对括号，有1种类型：

(a+b)+(c+d)

第二种，括号里有两个运算符，共6种：

只有一对括号，有2种类型：

(a+b+c)+d a+(b+c+d)

有两对括号，有4种类型：

(a+(b+c))+d ((a+b)+c)+d a+((b+c)+d) a+(b+(c+d))

第三种，括号里有三个运算符，共1种： (a+b+c+d)

因此，括号位置总共有11种。

总的可能数为：24 * 64 * 11 = 16896

生成后缀表达式

已有的中缀表达式为：

(a+b)+c+d
a+(b+c)+d
a+b+(c+d)
(a+b)+(c+d)
(a+b+c)+d
a+(b+c+d)
(a+(b+c))+d
((a+b)+c)+d
a+((b+c)+d)
a+(b+(c+d))
(a+b+c+d)

由于后缀表达式能体现出括号所要表现的优先级。将它们转成后缀表达式，不用再考虑括号，易于计算。

左边是中缀表达式，右边是对应的后缀表达式。

(a+b)+c+d ab+c+d+
a+(b+c)+d abc++d+
a+b+(c+d) ab+cd++
(a+b)+(c+d) ab+cd++
(a+b+c)+d ab+c+d+
a+(b+c+d) abc+d++
(a+(b+c))+d abc++d+
((a+b)+c)+d ab+c+d+
a+((b+c)+d) abc+d++
a+(b+(c+d)) abcd+++
(a+b+c+d) ab+c+d+

后缀表达式，去重整理后：

ab+c+d+
ab+cd++
abc++d+
abc+d++
abcd+++

只有5种了，还容易计算。

填充和计算后缀表达式

在5种后缀表达式中，在abcd的位置换成1到9的数字，在+的位置换上+-*/。

然后对后缀表达式求值，判断结果是否等于24.

用一位数组表示一种后缀表达式。按上面的格式填充出表达式。

填充的思路：

四个1-9的数字的数组A[0,4);
三个运算符op1,op2,op3;
vv[5][7]={
            {A[0],A[1],-op1,A[2],-op2,A[3],-op3},//  ab+c+d+
            {A[0],A[1],-op1,A[2],A[3],-op2,-op3},//  ab+cd++
            {A[0],A[1],A[2],-op1,-op2,A[3],-op3},//  abc++d+
            {A[0],A[1],A[2],-op1,A[3],-op2,-op3},//  abc+d++
            {A[0],A[1],A[2],A[3],-op1,-op2,-op3},//  abcd+++
        };

注意：运算符前加-，是用来区分运算符和数字的。

后缀表达式的计算，思路简洁。

遇到数字，数字入栈；遇到运算符，出栈两个数字，根据运算符类型计算值，值再入栈。

注意，本题要对五种后缀表达式，每一种表达式都要算一遍。

计算思路：

for(i=0;i<5;++i){
    空栈S;
    for(j=0;j<7;++j){
        if(vv[i][j]是数字){
            数字入栈S;
        }else{
            //运算符op
            出栈右操作数b;
            出栈右操作数a;
            根据运算符op的类型，计算+-*/
            c=a op b;
            c入栈S;
        }
    }
}

分数处理

由于本题中除号不再是整数除法，要引入分数。

引入分数类F，并重载了加减乘除四种运算符。

乘除，相对简单。除法要考虑除法不能为0。

加减，分母要用最小公倍数通分，分子再求加减。

加减乘除，最后一步都是用分子和分母的最大公约数去约分。

最大公约数算法：

int gcd(int a,int b){
    if(!b) return a;
    return gcd(b,a%b);
}

最小公倍数算法：

int lcd(int a,int b){
    return (a/gcd(a,b))*b;
}

gcd和lcd算法这里不再赘述。

分数类F：

struct F{
    分子up;
    分母dn;
    F(up=0,dn=1);
    重载运算符+;
    重载运算符-;
    重载运算符*;
    重载运算符/;
    compact();//约分
    is_zero();//是否为0
};

数字全排列和枚举运算符

四个数字的全排列，可以用回溯法去求。在这里，用next_permutation去求全排列。

枚举运算符，每种就四种，一个三层嵌套循环。

思路：

数组A;
sort(数组A);
do{
    for(op1={+,-,*,/}){
        for(op2={+,-,*,/}){
            for(op3={+,-,*,/}){
                数组A的一个排列和op1,op2,op3求值;
                如果值是24，返回成功;
            }
        }
    }
}while(next_permutation(数组A));

完整逻辑

算法整体框架：

    bool judgePoint24(vector<int>& nums) {
        const char ops[4]={'+','-','*','/'};
        sort(nums.begin(),nums.end());
        do{
            for(int i=0;i<4;++i){
                for(int j=0;j<4;++j){
                    for(int k=0;k<4;++k){
                        bool r=get_expr(nums,ops[i],ops[j],ops[k]);
                        if(r) return true;
                    }
                }
            }
        }while(next_permutation(nums.begin(),nums.end()));
        return false;
    }

后缀表达式的填充和计算：

    bool get_expr(vector<int>& A,char op1,char op2,char op3){
        const int R=5;
        const int C=7;
        int vv[R][C]={
            {A[0],A[1],-op1,A[2],-op2,A[3],-op3},
            {A[0],A[1],-op1,A[2],A[3],-op2,-op3},
            {A[0],A[1],A[2],-op1,-op2,A[3],-op3},
            {A[0],A[1],A[2],-op1,A[3],-op2,-op3},
            {A[0],A[1],A[2],A[3],-op1,-op2,-op3},
        };
        for(int i = 0;i<R;++i){
            stack<F> S;
            bool fail=false;
            for(int j = 0;j<C;++j){
                if(vv[i][j]>0){
                    S.push(F(vv[i][j]));
                }else{
                    F b=S.top();S.pop();
                    F a=S.top();S.pop();
                    if(!(a.is_valid() && b.is_valid())) {
                        fail=true;
                        break;
                    }
                    
                    F c;
                    switch(-vv[i][j]){
                        case '+':
                            c=a+b;
                            break;
                        case '-':
                            c=a-b;
                            break;
                        case '*':
                            c=a*b;
                            break;
                        case '/':
                            if(b.is_zero()) {
                                fail=true;
                                break;
                            }
                            c=a/b;
                            break;
                    };
                    if(fail || !c.is_valid()) {
                        fail=true;
                        break;
                    }
                    S.push(c);
                }
            }
            if(!fail){
                F c=S.top();
                if(c.is_integer() && c.up == 24) return true;
            }
            
        }
        return false;
    }

分数类F：

    struct F{
        int up;
        int dn;
        F(int _up = 0,int _dn = 1) : up(_up),dn(_dn){}
        F(const F& o){
            up = o.up;
            dn = o.dn;
        }
        //加减乘除运算符的重载
        F operator+(const F& o)const{
            F t(*this);
            if(dn == o.dn){
                t.up += o.up;
                t.compact();
            }else{
                int _lcd = lcd(dn,o.dn);
                int _up = (_lcd/dn) * up + (_lcd/o.dn) * o.up;
                int _gcd = gcd(_up,_lcd);
                t.up = _up / _gcd;
                t.dn = _lcd / _gcd;
            }
            
            return t;
        }

        F operator-(const F& o)const{
            F t(*this);
            if(dn == o.dn){
                t.up -= o.up;
                t.compact();
            }else{
                int _lcd = lcd(dn,o.dn);
                int _up = (_lcd/dn) * up - (_lcd/o.dn) * o.up;
                int _gcd = gcd(_up,_lcd);
                t.up = _up / _gcd;
                t.dn = _lcd / _gcd;
            }
            
            return t;
        }
    
        F operator*(const F& o)const{
            F t(*this);
            t.up = up*o.up;
            t.dn = dn*o.dn;
            t.compact();
            return t;
        }
        
        F operator/(const F& o)const{
            F t(*this);
            t.up = up * o.dn;
            t.dn = dn * o.up;
            t.compact();
            return t;
        }
        
        void compact(){
            int _gcd=gcd(up,dn);
            up = up/_gcd;
            dn = dn/_gcd;
        }
        
        string to_str()const{
            if(dn==1) return to_string(up);
            return to_string(up)+"/"+to_string(dn);
        }
        
        bool is_integer()const{//是不是整数
            return dn==1;
        }
        
        bool is_zero()const{//是不是0
            return up == 0 && dn !=0;
        }
        
        bool is_valid()const{//是不是0
            return dn != 0;
        }
    };

解法二 回溯法

解法一，将数字，运算符和括号都确定后，再填充后缀表达式，再求值。特点是先生成，再判断的过程。

也可以采用边生成，边判断的方式。

例如，[4, 1, 8, 7]

第一步，选择两个不同的元素a和b。共有12种。

比如a=4 b=1。

此时有加减乘除四种方法。

不妨选加法，求得c=a+b=4+1=5。

这就构成了一个新的序列，不再含有a和b，但是补上了c。

新序列[8,7,5]。

进一步要干啥？

递归，计算新序列。新序列少了1个元素，且计算方法是相同的。算法一定是收敛的。

如果选择减法，求得c=a-b=4-1=3。

有新序列[3,8,7]。再递归。

如果c=b-a=1-4=-3。

有新序列[-3,8,7]。再递归。

因此，算法的思路：

bool dfs(数组A){
    if(数组A 只有1个元素 且 A[0] == 24) 返回成功;
    for(a in 数组A){
        for(b in 数组A){
            if(a和b是同一个位置) continue;
            for(op = {+,-,*,/}){
                if(op == + 或 op == *){
                    if(a 和 b，已经算过1次加法和乘法) 不再重复计算,continue;
                    c = a op b;
                    数组left={数组A中除a和b的数字} + c;
                    r = dfs(数组left);
                    if(r 成功) 返回成功;
                }
                if(op == - 或者 op == /){
                    if(op == / 并且 b为0) 除法不做，continue;
                    c = a op b;
                    数组left={数组A中除a和b的数字} + c;
                    r = dfs(数组left);
                    if(r 成功) 返回成功;
                }
            }
        }
    }
}

要注意：

加法和乘法，满足交换律，只需算一次。

除法，要排除掉除数为0的情况；

完整的算法：

分数类F，同解法一。

    bool judgePoint24(vector<int>& nums) {
        vector<F> A(nums.size());
        for(int i=0;i<nums.size();++i){
            A[i] = F(nums[i]);
        }
        if(dfs(A)) return true;
        return false;
    }
    
    bool dfs(const vector<F>& A){
        if(A.empty()) return false;
        if(A.size() == 1 && A[0].is_integer() && A[0].up == 24) return true;
        for(int i=0;i<A.size();++i){
            for(int j=0;j<A.size();++j){
                if(i == j) continue;
                vector<F> left(A.size()-2+1);
                int t=0;
                for(int k=0;k<A.size();++k){
                    if(k != i && k != j){
                        left[t++] = A[k];
                    }
                }
                
                if(i < j){
                    left[t] = A[i]+A[j];
                    if(dfs(left)) return true;
                    left[t] = A[i]*A[j];
                    if(dfs(left)) return true;
                }
                left[t] = A[i]-A[j];
                if(dfs(left)) return true;
                if(A[j].is_zero()) continue;
                left[t] = A[i]/A[j];
                if(dfs(left)) return true;
            }
        }
        return false;
    }
