leetcode72 编辑距离 Edit Distance

解法一 动态规划

设s(i)表示串s中长度为i的前缀。

先考虑，怎么将word1(i)通过替换，插入和删除变成word2(j)。

例子：word1=horse, word2=ros。

制作一个表格d，word1的每个字符作为列，word2的每个字符作为行。单元格d(i,j)的值为将word2(i)变成word1(j)的最少步数。

首先将word2(i)变成空串word1(0)，那么word2(i)的字符个数就为其步骤数。

这就是图中的第0列的值。

同样，将空串word2(0)变成串word1(j)，那么，就构成图中的第0行。

可以发现d(i,j)的值只与d(i-1,j)，d(i-1,j-1)和d(i,j-1)有关。

d(i-1,j)——word2(i-1)变成word1(j)的最少步骤数。

d(i,j)的可能等于d(i-1,j)+1。因为至少要加1个字符word2[i-1]。

d(i,j-1)——word2(i)变成word1(j-1)的最少步骤数。

d(i,j)的可能等于d(i,j-1)+1。因为至少要加1个字符word1[j-1]。

d(i-1,j-1)——word2(i-1)变成word1(j-1)的最少步骤数。

那么，当word2[i-1] == word1[j-1]时，d(i,j)可以等于d(i-1,j-1)。

当word2[i-1] != word1[j-1]时，d(i,j)可以等于d(i-1,j-1)+1。

综上，d(i,j) = min(d(i-1,j)+1,d(i,j-1)+1,d(i-1,j-1) + word2[i-1] == word1[j-1] ? 1 : 0)。

逐行计算例子中的d(i,j)得出

算法为：

    int minDistance(string word1, string word2) {
        const int len1 = word1.size();
        const int len2 = word2.size();
        vector<vector<int>> d(len2+1,vector<int>(len1+1,0));
        
        for(int i = 0;i<=len2;++i){
            for(int j = 0;j<=len1;++j){
                if(i == 0 && j == 0){
                    d[i][j] = 0;
                }else if(i == 0){
                    d[i][j] = j;
                }else if(j == 0){
                    d[i][j] = i;
                }else{
                    d[i][j] = min(
                        min(d[i-1][j],d[i][j-1]) + 1, 
                        d[i-1][j-1] + ((word2[i-1] == word1[j-1]) ? 0 : 1));
                }
            }
        }
        return d[len2][len1];
    }

另外，算法是从左向右，从上向下。因此d(i-1,j-1),d(i-1,j),d(i,j-1)是依次计算的。

这样可想到，表格d有两行就足够了。

本题递推计算的特性，再进一步，可以用单行的滚动数组d即可完成。即去掉word2这一维，保留word1这一维。其它不变。

要注意的是，在更新d[j]时。d[j]对于位置d[j+1]来说，相当于二维时 d[i][j]对d[i+1][j+1]的作用。因此在更新d[j]前，应该要保存下d[j]的旧值。

因此，基于滚动数组的算法：

    int minDistance(string word1, string word2) {
        const int len1 = word1.size();
        const int len2 = word2.size();
        vector<int> d(len1+1,0);
        int pre_j_1 = 0;
        for(int i = 0;i<=len2;++i){
            for(int j = 0;j<=len1;++j){
                if(i == 0 && j == 0){
                    d[j] = 0;
                }else if(i == 0){
                    d[j] = j;
                }else if(j == 0){
                    pre_j_1 = d[j];
                    d[j] = i;
                }else{
                    int t= min(
                        min(d[j],d[j-1]) + 1, 
                        pre_j_1 + ((word2[i-1] == word1[j-1]) ? 0 : 1));
                    pre_j_1 = d[j];
                    d[j] = t;
                }
            }
        }
        return d[len1];
    }

解法二 递归法

解法一中已明确，d(i,j)的值只与d(i-1,j)，d(i-1,j-1)和d(i,j-1)有关。

可以递归先算出这三个值。而且配合表格d，还能做到不重复递归。

    int minDistance(string word1, string word2) {
        const int len1 = word1.size();
        const int len2 = word2.size();
        vector<vector<int>> d(len2+1, vector<int>(len1+1,-1));
        return dfs(d,word2,len2,word1,len1);
    }
    
    int dfs(vector<vector<int>>& d,
            const string& word2, int i, const string& word1, int j){
        int& ans = d[i][j];
        if(ans > -1) return ans;
        
        if(i == 0 && j == 0){
            ans = 0;
        }else if(i == 0){
            ans = j;
        }else if(j == 0){
            ans = i;
        }else{
            int a=dfs(d,word2,i-1,word1,j-1);
            int b=dfs(d,word2,i-1,word1,j);
            int c=dfs(d,word2,i,word1,j-1);
            ans = min(min(b,c)+1,a+((word2[i-1] == word1[j-1]) ? 0 : 1));
        }
        return ans;
    }
