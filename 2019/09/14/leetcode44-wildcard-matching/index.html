leetcode44 通配符匹配 Wildcard Matching

解法一 回溯法

本题，用回溯法解决，细节多，坑点多。尽管如此，写出正确的算法，到也不太难。难点在于正确的减枝，优化时间复杂度。整体看，回溯法较复杂。

回溯法的整体思路是，串p的字符从左向右，匹配串s中的字符，

如果p[i] == s[j] || p[i] == '?'时，递归匹配p[i+1,end]和s[j+1,end]。

如果p[i] == ‘*’时，可以递归匹配空串，s[j , end],s[j+1 , end ],s[j+2 , end ]....。 即，匹配s串的全部后缀串，及空串。

为了简便，设串s的长度为slen，p串的长度为plen。

简化模式串p

p中连续多个*，其作用等同于单个*。首先将p中连续多个*，简化为单个*。

    string remove(const string& s){
        string t;
        for(int i=0;i<s.size();++i){
            if(s[i] != '*'){
                t += s[i];
            }else{
                if(t.empty() || t.back() != '*') t += s[i];
            }
        }
        return t;
    }

递归边界条件

1，当 slen == 0 且 plen == 0，空串匹配空串，匹配成功。

2，当 slen == 0 且 plen > 0， 此时要分情况。

只要串p中，有不是*的字符，匹配失败。

否则成功。

例如：s="", p="*a*b"。

匹配失败。

s="", p="*"

匹配成功。

3，当 slen>0 且 plen == 0 时，匹配失败。但是在回溯框架下，这种情况，还可以回溯到前一次，再递归匹配。

例如：s="abc",p=""。

匹配失败。

4，当 plen == 1 且 p[0]=='*'时，匹配成功。*可以匹配任何串。

例如：s="fadawgag",p="*"。

匹配成功。

5，当 s 和 p 的开始一部分匹配时，则递归匹配s和p的剩余部分。

例如：s="abcdxyfdafa" , p = "a?cdgaaga"。

那么，s开始的四个字符abcd，与p开始的四个字符a?cd，匹配。s和p剩余的部分再递归匹配。

6，当s和p第一对字符就不匹配 且 p第一个字符不是*，匹配失败。 但是在回溯框架下，这种情况，还可以回溯到前一次，再递归匹配。

例如：s="abcd"，p="kabcd"。

匹配失败。

7，当p的第一个字符，p[0]=='*'时，要匹配s串的全部后缀串，及空串。

思路整合为：

int match(串s,串p){
    s长度，slen;
    p长度，plen;
    //条件1
    if(slen == 0 && plen == 0) 返回 匹配成功;
    //条件2
    if(slen == 0 && plen){
        if(p中有一个不是*的字符) 返回 匹配失败，不再回溯;
        else 返回 匹配成功;
    }
    //条件3
    if(slen && plen == 0) 返回 匹配失败，再回溯;
    //条件4
    if(plen == 1 && p[0] == '*') 返回 匹配成功;
    //条件5
    计算s和p完全ge匹配(不含*的部分)的前缀的最大长度，len;
    if(len > 0) 返回 递归匹配 match(s[len,slen],p[len,plen]);
    //条件6
    if(p[0] != '*') 返回 匹配失败，再回溯;
    //条件7
    //此时p[0] == '*'，p[1]为*后面的第一个字符
    for(i=0;i<slen;++i) {
        //后缀i的第一个字符与p[1]不匹配时，减枝
        if(s[i] != p[1] && p[1] != '?') continue;
        r = match(后缀i+1,p[2,plen]);
        if(r 不是再回溯) 返回r;
    }
    //条件8
    返回 匹配失败，再回溯;
}

将回溯失败区分成两种。

条件1和条件4，返回匹配成功，容易理解。

为什么，条件3，6和8，匹配失败，再回溯。而条件2，匹配失败，不再回溯了？

先看条件2，如果p中位置k处的字符不为*，p[k] != '*'，那么s已经结束了，p[k]无法匹配空串s。

出现这种情况，第一种可能是，s和p的输入就是这样的。结果匹配失败。

第二种可能是，s和p是某levelA层递归的直接下一层或间接下一层，叫levelB。在levelA层中，由条件7引起的。而在条件7中，每次递归传入的p[2,plen]是固定的。再次递归时，肯定会出现与层levelB相同的现象，而再次触发条件2。因此，条件2时，不用再回溯。这相当于回溯减枝。

条件3，6，8，也可能是某levelA层 递归的直接下一层或间接下一层，叫levelB。

条件3和条件8，可能由levelA中的条件7引起。再一次回溯，s的后缀串会发生变化，结果可能不同。

条件6，不可能是levelB直接上层中的条件7引起，但可能是levelA的条件7引起。回溯再递归，结果可能变化。

完整的AC代码：返回值：1——匹配成功，0——匹配失败，-1——再回溯。

    bool isMatch(string s, string p) {
        string pp = remove(p);
        return isMatch2(s,pp) == 1;
    }

    int isMatch2(string s, string p) {
        const int slen = s.size();
        const int plen = p.size();
        if(slen == 0 && plen == 0) return 1;
        if(slen == 0 && plen){
            for(int i=0;i<plen;++i){
                if(p[i]!='*') return 0;
            }
            return 1;
        }
        if(slen && plen == 0) return -1;
        
        if(plen == 1){
            if(p[0] == '*') return 1;
        }
        
        int i;
        for(i=0;i<min(slen,plen);++i){
            if(s[i] != p[i] && p[i] != '?'){
                break;
            } 
        }
        if(i) {
            return isMatch2(s.substr(i),p.substr(i));
        }
        
        if(p[0] != '*') return -1;
        
		string p_rest = p.substr(1);
        string prp = p_rest.substr(1);
        int r;
        for(int i=0;i < slen;++i){
            if(s[i] != p_rest[0] && p_rest[0] != '?') continue;
            string ss = s.substr(i+1);
			if( (r=isMatch2(s.substr(i+1),prp)) != -1 ) return r;
        }
        return -1;
    }

        for(int i=0;i<s.size();++i){

                t += s[i];
            }else{
                if(t.empty() || t.back() != '*') t += s[i];
            }
        }
        return t;

重要的减枝

考虑这种情况。

p中字符*的个数为star_cnt。p去除字符*的长度为:p-star_cnt。 p去除字符*后的 串为q。

如果 串s的长度slen < p- star_cnt时，即串s比串q短。此时，不论用什么方法，s和p一定匹配失败。因此，可以减枝。

而p串及其后缀串中的*数量，可以实现算出来。

设数组pp_star_cnt，pp_star_cnt[i]——p的后缀p[i,plen]中字符*的个数。

我们可以从后往前遍历p，并更新pp_star_cnt数组。

        string pp = remove(p);
        vector<int> pp_star_cnt(pp.size()+1,0);
        int cnt=0;
        for(int i=pp.size()-1;i>=0;--i){
            if(pp[i] == '*') ++cnt;
            pp_star_cnt[i] = cnt;
        }

减枝条件可简化为：

slen < plen - pp_star_cnt[i]

加了减枝的AC代码：

    bool isMatch(string s, string p) {
        string pp = remove(p);
        vector<int> pp_star_cnt(pp.size()+1,0);
        int cnt=0;
        for(int i=pp.size()-1;i>=0;--i){
            if(pp[i] == '*') ++cnt;
            pp_star_cnt[i] = cnt;
        }
        return isMatch2(s,pp,pp_star_cnt) == 1;
    }

    int isMatch2(string s, string p,const vector<int>& pp_star_cnt) {
        const int slen = s.size();
        const int plen = p.size();
        const int pp_star_len = pp_star_cnt.size();
        if(slen < plen - pp_star_cnt[pp_star_len - 1 - plen]) return 0;
        if(slen == 0 && plen == 0) return 1;
        if(slen == 0 && plen){
            for(int i=0;i<plen;++i){
                if(p[i]!='*') return 0;
            }
            return 1;
        }
        if(slen && plen == 0) return -1;
        
        if(plen == 1){
            if(p[0] == '*') return 1;
        }
        
        int i;
        for(i=0;i<min(slen,plen);++i){
            if(s[i] != p[i] && p[i] != '?'){
                break;
            } 
        }
        if(i) {
            return isMatch2(s.substr(i),p.substr(i),pp_star_cnt);
        }
        
        if(p[0] != '*') return -1;
        
		string p_rest = p.substr(1);
        string prp = p_rest.substr(1);
        const int prp_min_len = prp.size() - pp_star_cnt[pp_star_len - 1 - prp.size()];
        int r;
        for(int i=0;i < slen;++i){
            if(s[i] != p_rest[0] && p_rest[0] != '?') continue;
            string ss = s.substr(i+1);
            if(ss.size() < prp_min_len) return 0;
			if( (r=isMatch2(s.substr(i+1),prp,pp_star_cnt)) != -1 ) return r;
        }
        return -1;
    }

                t += s[i];
            }else{
                if(t.empty() || t.back() != '*') t += s[i];
            }
        }
        return t;

解法二 动态规划算法

可以看出 ，解法一，匹配一对字符后，再用递归解决s和p的后缀。如果后缀匹配，则结果也匹配。

可以尝试，将这种思路转换为递推算法。

从后往前递推算法

设s(i)表示 s串从i开始的后缀 。p(i)表示 p串从j开始的后缀 。

slen 和 plen表示串s和p的长度

设d[i][j]表示s(i)与p(j)是否匹配。

显然有 s(slen) 和 p(plen)为空，则匹配。

d[slen][plen] = 1。

当 s(slen) 但 j < plen，p(j)不为空时。此时仅当p(j)中全是字符*时，才匹配。

d[slen][j] = p[j] == '*' && d[slen][j+1]。

当 i < slen，s(i)不为空 且 p(plen)时。此时s(i)要与空串匹配，都不匹配。

d[i][plen] = 0。

当 i < slen 且 j < plen时，

当 s[i] == p[j] || p[j] == '?' 时。如果s(i+1)与p(j+1)匹配，则匹配。

d[i][j] = d[i+1][j+1]。

当 p[j] == '*' 时。

如果p[j]匹配s中的0个字符，此时就等价于p[j]这个位置不存在，匹配结果由p(j+1)决定。

此时 d[i][j] = d[i][j+1]。

如果p[j]匹配s中的1个字符，即p[j]匹配s[i]。那么匹配结果等价于p(j)匹配s(i+1)，但不能等价于p(j+1)匹配s(i+1)。

这个区别很关键。这是为什么了？

因为 等价于p(j+1)匹配s(i+1)，存在一种错误的情况。

当 p(j+1)匹配s(i+1)匹配时，p(j)和s(i)一定匹配。这一部分是正确的。

但是，当 p(j+1)与s(i+1) 不匹配时，不能断定 p(j)和s(i)一定不匹配 。因为字符*还可以匹配连续的2个，3个,...等连续任意个字符。可能在匹配k个字符时，匹配成功了。

如果 p[j]匹配s中的不止1个连续字符，比如匹配k个时，那么，按道理应该要考虑p(j)和s(i+k)。

实其不用，s(i+k)与p(j), s(i+k - 1)与p(j),..., s(i+1)与p(j) 这些匹配情况，在s(i)与p(j)匹配之前就已经算出过了。在位置i考虑 s(i+k)与p(j), 同 在位置i+k-1处，考虑 s(i+k-1+1)与p(j), 结果相同。

因此只需考虑s(i+1)与p(j)。

思路整理成代码：

    bool isMatch(string s, string p) {
        const int slen = s.size();
        const int plen = p.size();
        //d[i][j]表示s串从i开始的后缀与p串从j开始的后缀是否匹配
        vector<vector<int>> d(slen+1,vector<int>(plen+1,0));

        for(int i=slen;i>=0;--i){
            for(int j=plen;j>=0;--j){
                if(i == slen && j == plen) {
                    d[i][j] = 1;
                    continue;
                }else if(i == slen){
                    d[i][j] = (p[j] == '*' && d[i][j+1]) ? 1 : 0;
                }else if(j == plen){
                    d[i][j] = 0;
                }else{
                    if(s[i] == p[j] || p[j] == '?'){
                        d[i][j] = d[i+1][j+1];
                    }else if(p[j] == '*'){
                        d[i][j] = d[i][j+1] || d[i+1][j];
                    }else{
                        d[i][j] = 0;
                    }
                }
            }
        }
        
        return d[0][0];
    }

从前往后递推算法

同样，如果考虑从串开始到结束来匹配，即前缀匹配。递推算法就会从前往后。

设s(i)为串s的长度为i的前缀， p(i)为串p的长度为i的前缀 。

d[i][j]表示s(i)与p(j)是否匹配。

要注意的是s(0)与p(0)，现在为空串了。

还要注意，当i>0时，s(i)的最后一个字符为s[i-1]。当j>0时，p(j)的最后一个字符为p[j-1]。

只要对上面的递推算法改为从前往后递推即可。

    bool isMatch(string s, string p) {
        const int slen = s.size();
        const int plen = p.size();
        //d[i][j]表示s串长度为i的前缀与p串长度为j的前缀是否匹配
        vector<vector<int>> d(slen+1,vector<int>(plen+1,0));

        for(int i=0;i<=slen;++i){
            for(int j=0;j<=plen;++j){
                if(i == 0 && j == 0) {
                    d[i][j] = 1;
                    continue;
                }else if(i == 0){
                    d[i][j] = (p[j-1] == '*' && d[i][j-1]) ? 1 : 0;
                }else if(j == 0){
                    d[i][j] = 0;
                }else{
                    if(s[i-1] == p[j-1] || p[j-1] == '?'){
                        d[i][j] = d[i-1][j-1];
                    }else if(p[j-1] == '*'){
                        d[i][j] = d[i][j-1] || d[i-1][j];
                    }else{
                        d[i][j] = 0;
                    }
                }
            }
        }
        
        return d[slen][plen];
    }
