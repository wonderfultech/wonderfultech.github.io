解法一 递归法（超时）

最后一个case超时了。

思路，每次找表达式中优先级最低或最后计算的运算符c，递归计算字符c分割的前后两部分。这样递归，就形成了一颗表达式树。字符c就是根。

例如： "1 + 1"

最后计算的字符为+。构成树。

    +
1        1

例如：" 2-1 + 2 "

最后计算的字符为+。

        +
    -       2
2       1

例如： "(1+(4+5+2)-3)+(6+8)"

最后计算的字符也是+。

                +
        -                 +
    +         3       6       8
1        +         
    +       2
4       5

怎么找出优先级最低的运算符呢？

首先，括号内的运算符不是优先级最低的运算符。要找括号外的运算符。

比如:"(2-1)+2"

-不是最后计算的运算符。

其次，+和-，是左结合的运算符。相同优先级时，最右边的运算符是最后计算。

找最后计算的运算符的思路：

串s;
last_operator = -1;
p：0 - 表示在括号外；大于0 - 表示在几层括号内；
for(i in [x,y)){
    switch(s[i]){
         case '(': ++p; break;
         case ')': --p; break;
         case '+':
         case '-':
         if(!p) last_operator = i;
         break;
    }
}

注意:

用p跳过括号内的内容；

last_operator会被多次更新，最终的字符，就是最后运算符。

如果last_operator==-1，表示没有找到最后计算的运算符。

此时有两种情况，

要么，表达式被一个大括号括住。

此时，去掉括号，递归调用。

要么，表达式是数字。

提取数字。

完整的思路：

//[x,y)
dfs(s,x,y){
    last_operator = -1;
    p：0 - 表示在括号外；大于0 - 表示在几层括号内；
    for(i in [x,y)){
        switch(s[i]){
         case '(': ++p; break;
         case ')': --p; break;
         case '+':
         case '-':
         if(!p) last_operator = i;
         break;
        }
    }
    if(last_operator<0){
        if(s[x] == '(') 递归dfs(s,x+1,y-1),返回；
        提取数字，返回；
    }
    left = 递归dfs(s,x,last_operator);
    right = 递归dfs(s,last_operator+1,y);
    if(s[last_operator] == '+') 返回left+right;
    if(s[last_operator] == '-') 返回left-right;
}

完整的代码：

    typedef long long ll;
    int calculate(string s) {
        string t;
        for(auto x : s){
            if(x!=' '){
                t+=x;
            }
        }
        return dfs(t,0,t.size());
    }

    ll dfs(const string & s,int x,int y){
        int last_operator = -1;
        int p = 0;
        for(int i = x;i < y;++i){
            switch(s[i]){
                case ')':++p;
                break;
                case '(':--p;
                break;
                case '+':
                case '-':
                if(!p) last_operator = i;
                break;
                default:
                break;
            }
        }

        if(last_operator < 0){
            if(s[x]=='('){//最外面的是括号
                return dfs(s,x+1,y-1);
            } 
            //没有运算符也没有括号，是整数
            ll r=0;
            for(int i=x;i<y;++i){
                r = 10*r + (s[i] - '0');
            }
            return r;
        }
        ll left = dfs(s,x,last_operator);
        ll right = dfs(s,last_operator+1,y);
        ll r;
        switch(s[last_operator]){
            case '+': r = left + right;
            break;
            case '-': r = left - right;
            break;
        }
        return r;
    }

解法二 双栈法

一个运算符栈opors，一个运算数栈opands。

固定方法。

括号内的计算优先级最高。

碰到'('，'('入opors栈。

碰到')'，opors栈中栈顶到'('的运算符都要计算出来。 这表示括号内的表达式结束了。然后， '(' 出栈。

碰到'+''-'， opors栈中栈顶到'('的运算符都要计算出来。 这是左结合性导致的。然后 '+''-' 入栈。

碰到数字，提取出整数，然后入opands栈。

至此，怎么按opors计算数值呢？

opors计算逻辑：


while(opors不为空 且 opors栈顶是'+'或'-'){
    b = opands出栈;
    a = opands出栈;
    op = opors出栈;
    r = a op b;
    r入opands栈;
}

注意，字符串遍历完后，可能会存在opors中还有运算符的情况。

比如:"1+2"

就会出现opors中为'+'，而opands中为1,2的情况。

此时，要对opors中的字符，逐一完成计算。

完整的思路：

for(i in s){
    if(s[i]是'(') '('入栈opors;
    if(s[i]是')') 执行opors计算逻辑; '('出栈;
    if(s[i]是'+'或'-') 执行opors计算逻辑; '+'或'-'入栈;
    if(s[i]是数字) 提取整数,入栈opands;
}
执行opors计算逻辑;

完整代码：

    typedef long long ll;
    int calculate(string s) {
        string t;
        for(auto x : s){
            if(x!=' '){
                t+=x;
            }
        }
        return done(t,0,t.size());
    }

    ll done(const string & s,int x,int y){
        stack<ll> opands;
        stack<char> opors;
        int p = x;
        while(p < y){
            if(s[p] == '('){
                opors.push(s[p]);
                ++p;
            }else if(s[p] == '+' || s[p] == '-' || s[p]==')'){
                while(opors.size() && (opors.top() == '+' || opors.top() == '-')){
                    ll b = opands.top();opands.pop();
                    ll a = opands.top();opands.pop();
                    char op = opors.top();opors.pop();
                    if(op == '+'){
                        opands.push(a+b);
                    }else if(op == '-'){
                        opands.push(a-b);
                    }
                }
                if(s[p] == ')'){
                    if(opors.size() && opors.top()=='(') opors.pop();
                }else{
                    opors.push(s[p]);
                }
                ++p;
            }else{
                ll r=0;
                while(p<y && isdigit(s[p])){
                    r = 10 * r + (s[p++] - '0');
                }
                opands.push(r);
            }
        }
        while(opors.size()){
            ll b = opands.top();opands.pop();
            ll a = opands.top();opands.pop();
            char op = opors.top();opors.pop();
            if(op == '+'){
                opands.push(a+b);
            }else if(op == '-'){
                opands.push(a-b);
            }
        }
        return opands.top();
    }

解法三 递归双栈法

在解法二的基础上修改，

当碰到'('时，递归求解括号内的表达式。求解的结果入opands栈。

而，碰到')'时，括号内的表达式遍历完成，结束遍历。完成opors中运算符计算后，再返回。

其它与解法二完全一致。

完整的逻辑：

dfs(s,x,y下一位置next){
    for(i in s){
        if(s[i]是'(') r=dfs(s,x+1,last);r入opands栈;i = last;
        if(s[i]是')') next=i+1;不再遍历;
        if(s[i]是'+'或'-') 执行opors计算逻辑; '+'或'-'入栈;
        if(s[i]是数字) 提取整数,入栈opands;
    }
    执行opors计算逻辑;
    返回值;

完整代码：

    typedef long long ll;
    int calculate(string s) {
        string t;
        for(auto x : s){
            if(x!=' '){
                t+=x;
            }
        }
        int next;
        return dfs(t,0,t.size(),next);
    }

    ll dfs(const string & s,int x,int y,int& next){
        stack<ll> opands;
        stack<char> opors;
        int p = x;
        while(p < y){
            if(s[p] == '('){
                int last;
                ll r=dfs(s,p+1,y,last);
                opands.push(r);
                p = last;
            }else if(s[p] == '+' || s[p] == '-'){
                while(opors.size() && (opors.top() == '+' || opors.top() == '-')){
                    ll b = opands.top();opands.pop();
                    ll a = opands.top();opands.pop();
                    char op = opors.top();opors.pop();
                    if(op == '+'){
                        opands.push(a+b);
                    }else if(op == '-'){
                        opands.push(a-b);
                    }
                }
                opors.push(s[p]);
                ++p;
            }else if(s[p] == ')'){
                next=++p;
                break;
            }else{
                ll r=0;
                while(p<y && isdigit(s[p])){
                    r = 10 * r + (s[p++] - '0');
                }
                opands.push(r);                
            }
        }
        while(opors.size()){ 
            ll b = opands.top();opands.pop();
            ll a = opands.top();opands.pop();
            char op = opors.top();opors.pop();
            if(op == '+'){
                opands.push(a+b);
            }else if(op == '-'){
                opands.push(a-b);
            }
        }
        return opands.top();
    }

解法四 单栈法

只有+和-，可以将-与后面的数字合在一起考虑，引入数字的符号。这样整个表达式都是加法。

而且+和-都是从左向右结合的。这样的随着表达式从左向右推进，每遇到+和-，就将刚取出的整数，与前面的值累加。

例如："2-1+2"

假设现在遍历到1的位置。此时左边已算出的值:left = 2。当前的数字：next_integer = 1，其符号是 next_sign = -1。

往右推进到+，此时将next_sign * next_integer 加到 left 上。

left = left + next_sign * next_integer = 2 -1 =1。

然后，+号，下一个数字的符号是正，next_sign = 1。

继续往右推进，直到结束。只要表达式中没有括号，就可以用这样的方式进行，并能得出正确的值。

唯一难处理的是括号。

这是引入栈。

当遇到'('时，要先算括号中的表达式，可将左边已经算出的值left和当前数字符号next_sign入栈。

当遇到')'时，括号中的表达式结束了。要将括号内的表达式的值，先求出。其符号和括号前的值，都在栈中。只要将它们取出，算出新的值。

例如："1-(2-3)+6"

当遍历到'('时，left=1,next_sign = -1。入栈S=[1,-1]。

当遇到')'时，括号内的值为-1。

此时，要与之前的值整合。

出栈。

之前的值是1。

括号内的值的符号是-1。

整合值为1+(-1 * -1) = 2。

完整的思路：

left = 0: 已算出的值;
next_sign = 1: 下一部分的符号;
next_integer = 0: 下一部分的值;
S：栈;
for (i in s){
    if(s[i] 是 '(') left入栈S; next_sign入栈S;
    if(s[i] 是 ')') 合计出括号内的值；扩号前的值和括号值的符号出栈S，整合成新值;
    if(s[i] 是 '+' 或 '-') 合计出符号前的值;更新next_sign；
    if(s[i] 是 数字) 取整数;
}

完整的代码：

    typedef long long ll;
    int calculate(string s) {
        string t;
        for(auto x : s){
            if(x!=' '){
                t+=x;
            }
        }
        return done(t);
    }

    ll done(const string & s){
        stack<ll> S;
        ll left = 0;
        ll next_sign = 1;
        ll next_integer = 0;
        int i = 0;
        while(i < s.size()){
            if(s[i] == '('){
                S.push(left);
                S.push(next_sign);
                next_sign = 1;
                next_integer = 0;
                left = 0;
            }else if(s[i] == ')'){
                left = left + next_sign * next_integer;
                ll my_sign = S.top();S.pop();
                ll prev_left = S.top();S.pop();
                left = prev_left + my_sign * left;
                next_sign = 1;
                next_integer = 0;
            }else if(s[i] == '+' || s[i] == '-'){
                left = left + next_sign * next_integer;
                next_sign = s[i] == '+' ? 1 : -1;
                next_integer = 0;
            }else{
                while(i < s.size() && isdigit(s[i])){
                    next_integer = 10* next_integer + (s[i++] - '0');
                }
                --i;
            }
            ++i;
        }
        
        return left + next_sign * next_integer;
    }
