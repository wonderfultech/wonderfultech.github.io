解法一 回溯法

回溯法的思路不难想。

先考虑几种简单情况。

当s1为空时，比较s2==s3即可；

当s2为空时，比较s1==s3即可；

当 s1的长度 + s2的长度 != s3的长度 时，匹配失败；

复杂的情况

设s1(i)表示s1中从位置i开始的后缀串。s2(j)和s3(k)，同理。

当s1的前k个字符与s3的前k个字符匹配时，要递归考虑除去匹配部分后的子串。即递归检查s1(i),s2和s3(i)。

for(i < k){
    r = 递归检查s1(i),s2,s3(i);
    if(r) 返回检查成功;
}

而当s2的前k个字符与s3的前k个字符匹配时，用同样的方式递归检查。

减枝

在递归检查时，重复出现s1,s2和s3，只需计算一次，结果记忆化存储。

为了方便存储，可以构造一个哈希表。

hash(s1+";"+s2+";"+s3,值)。

键是s1,s2,s3的拼接串，便于计算。值是s1,s2和s3的匹配结果。

那么，完整的思路：

bool dfs(s1,s2,s3){
    key = s1+";"+s2+";"+s3;
    if(hash有key) 返回hash[key];
    if(s1空) 返回s2==s3,更新hash[key];
    if(s2空) 返回s1==s3,更新hash[key];
    if(s1.length + s2.length != s3.length) 更新hash[key]=false;
    for(i < min(s1.length,s3.length) 并且 s1[i] == s3[i]){
        r = dfs(s1(i+1),s2,s3(i+1));
        if(r) 更新hash[key]=true，返回;
    }
    for(i < min(s2.length,s3.length) 并且 s2[i] == s3[i]){
        r = dfs(s1,s2(i+1),s3(i+1));
        if(r) 更新hash[key]=true，返回;
    }
    更新hash[key]=false，返回;
}

完整代码：

    bool isInterleave(string s1, string s2, string s3) {
        unordered_map<string,bool> d;
        return dfs(s1,s2,s3,d);
    }
    
    bool dfs(string s1,string s2,string s3,unordered_map<string,bool>& d){
        string t=s1+";"+s2+";"+s3;
        if(d.count(t)) return d[t];
        bool& ans=d[t];
        if(s1.empty()) return ans = s2 == s3;
        if(s2.empty()) return ans = s1 == s3;
        if(s1.size() + s2.size() != s3.size()) return ans = false;
        for(int i=0;i<min(s1.size(),s3.size()) && (s1[i] == s3[i]);++i){
            bool r=dfs(s1.substr(i+1),s2,s3.substr(i+1),d);
            if(r) return ans = true;
        }
        for(int i=0;i<min(s2.size(),s3.size()) && (s2[i] == s3[i]);++i){
            bool r=dfs(s1,s2.substr(i+1),s3.substr(i+1),d);
            if(r) return ans = true;
        }
        return ans = false;
    }

解法二 动态规划

解法一，回溯法，递归判断s1,s2,s3。要明确一点，递归中检测的s1,s2,s3一定是原始串的后缀串。

那么，可以倒过来解决问题。

设s3(i)为以位置i开始的后缀串。

设当前有一组s1和s2的后缀串与s3(i+1)匹配。这组后缀串，包含了s1和s2后缀中，匹配s3(i+1)的全部情况。

这组匹配的s1和s2后缀串有如下形式：

s1(j_1) s2(k_1) 
s1(j_2) s2(k_2)
...
s1(j_n) s2(k_n)

满足条件：
j_1 < j_2 < ... < j_n;
k_1 > k_2 > ... > k_n;
s1.length - j_1 + s2.length - k_1 == s3.length - (i+1);
s1.length - j_2 + s2.length - k_2 == s3.length - (i+1);
...
s1.length - j_n + s2.length - k_n == s3.length - (i+1);

那么，从s3(i+1)的匹配情况，如何能得出s3(i)的匹配情况了？

注意此时S3后缀串其实变长了。

此时，借助解法1的思路，枚举s2(j)。固定s2(j)后,s1(k)也固定。

然后逐个匹配s1(k)和s3(i)的前t个相同字符。借助s3(i+1)的匹配情况，更新s3(i)的匹配情况。

从后往前递推

s3(i):s3从位置i开始的后缀;
s3后缀长度:s3_suffix_len = s3.length - i;
s2后缀长度:s2_suffix_len = min(s2.length, s3_suffix_len);
s1后缀长度:s1_suffix_len;
d[i][j][k]表示s3(i)与s2(j),s1(k)的匹配情况;
for(j = s2.length - s2_suffix_len  ; j <= s2.length){
    s1_suffix_len = min(s1.length, s3_suffix_len - (s2.length - j));
    c = i;
    //逐个比较s3(i)与s1(k)前部相同的字符
    for(k = s1.length - s1_suffix_len ;  k < s1.length && c < s3.length){
        if(s1[k] == s3[c]){
            d[c][j][k] = d[c][j][k] || d[c+1][j][k+1];
        }else break;
    }
}

同样， 枚举s1(k)。固定s1(k)后,s2(j)也固定。 可以得出类似上面的递推逻辑。

最终的结果是d[0][0][0]的值。

完整的代码不难写出。

    bool isInterleave(string s1, string s2, string s3) {
        const int len1=s1.size();
        const int len2=s2.size();
        const int len3=s3.size();
        if(len1 == 0) return s2 == s3;
        if(len2 == 0) return s1 == s3;
        if(len1 + len2 != len3) return false;
        vector<vector<vector<int>>> d(len3+1,vector<vector<int>>(len2+1,vector<int>(len1+1,0)));
        d[len3][len2][len1]=1;
        for(int i=len3;i>=0;--i){
            const int s3_suffix_len = len3 - i;
            const int s2_suffix_len = min(s3_suffix_len, len2);
            for(int j = len2 - s2_suffix_len; j <= len2; ++j){
                const int s1_suffix_len = min(len1, s3_suffix_len - (len2 - j));
                for(int k = len1 - s1_suffix_len, c = i; k < len1 && c < len3; ++k,++c){
                    if(s1[k] == s3[c]){
                        d[c][j][k] = d[c][j][k] || d[c+1][j][k+1];
                    }else break;
                }
            }
            const int s1_suffix_len = min(s3_suffix_len, len1);
            for(int j = len1 - s1_suffix_len; j <= len1; ++j){
                const int s2_suffix_len = min(len2, s3_suffix_len - (len1 - j));
                for(int k = len2 - s2_suffix_len, c = i; k < len2 && c < len3; ++k,++c){
                    if(s2[k] == s3[c]){
                        d[c][k][j] = d[c][k][j] || d[c+1][k+1][j];
                    }else break;
                }
            }
        }
        return d[0][0][0];
    }

降低到二维

从上面的分析看，只要s3(i)和s2(j)固定后，s1(k)必固定。

这意味着，此时只要比较字符s3[i]与s1[k]是否相等。如果s3[i] == s1[k]成立，此时可用d[i][j][k+1]的值更新d[i][j][k]。其它字符完全不用再比较。

同样， 只要s3(i)和s1(j)固定后，s2(k)必固定。

如果s3[i] == s2[k]成立，此时可用d[i][k+1][j]的值更新d[i][k][j]。

因此，可得出，d只需要两维即可。d中表示s3(i)的第一维度可以去掉。

新递推逻辑

s3(i):s3从位置i开始的后缀;
s3后缀长度:s3_suffix_len = s3.length - i;
s2后缀长度:s2_suffix_len = min(s2.length, s3_suffix_len);
s1后缀长度:s1_suffix_len;
d[j][k]表示s3(i)与s2(j),s1(k)的匹配情况;
for(j = s2.length - s2_suffix_len  ; j <= s2.length){
    s1_suffix_len = min(s1.length, s3_suffix_len - (s2.length - j));
    //比较s3[i]与s1[k]相同
    if(s1[k] == s3[i]){
        d[j][k] = d[j][k] || d[j][k+1];
    }
}

最终结果依然是d[0][0]。

完整的代码：

    bool isInterleave(string s1, string s2, string s3) {
        const int len1=s1.size();
        const int len2=s2.size();
        const int len3=s3.size();
        if(len1 == 0) return s2 == s3;
        if(len2 == 0) return s1 == s3;
        if(len1 + len2 != len3) return false;
        vector<vector<int>> d(len2+1,vector<int>(len1+1,0));
        d[len2][len1]=1;
        for(int i=len3;i>=0;--i){
            const int s3_suffix_len = len3 - i;
            const int s2_suffix_len = min(s3_suffix_len, len2);
            for(int j = len2 - s2_suffix_len; j <= len2; ++j){
                const int s1_suffix_len = min(len1, s3_suffix_len - (len2 - j));
                const int k = len1 - s1_suffix_len;
                if(s1[k] == s3[i]) d[j][k] = d[j][k] || d[j][k+1];
            }
            const int s1_suffix_len = min(s3_suffix_len, len1);
            for(int j = len1 - s1_suffix_len; j <= len1; ++j){
                const int s2_suffix_len = min(len2, s3_suffix_len - (len1 - j));
                const int k = len2 - s2_suffix_len;
                if(s2[k] == s3[i]) d[k][j] = d[k][j] || d[k+1][j];
            }
        }
        return d[0][0];
    }

从前往后递推

基于上面，对二维从后往前的递推逻辑的分析。用类似的方式，考虑从前往后递推分析。

设s3(i)表示串s3的长度为i的前缀。s3(i)的最后一个字符为s3[i - 1]。

s2(j)和s1(k)同理。

固定s3(i)，再固定s2(j)，那么就固定了s1(k)。此时比较s1[k - 1] 与 s3[i - 1]。如果 s1[k - 1] == s3[i - 1] 成立，则可以用d[j][k - 1] 更新 d[j][k]。

同理，固定s3(i)，再固定s1(j)，那么就固定了s2(k)。此时比较s2[k - 1] 与 s3[i - 1]。如果 s2[k - 1] == s3[i - 1] 成立，则可以用d[k - 1][j] 更新 d[k][j]。

从前往后推的逻辑：

s3(i):s3长度为i的前缀串;
s3前缀长度:s3_prefix_len = i;
s2前缀长度:s2_prefix_len = min(s2.length, s3_prefix_len);
s1前缀长度:s1_prefix_len;
d[j][k]表示s3(i)与s2(j),s1(k)的匹配情况;
for(j = 0  ; j <= s2_prefix_len){
    s1_prefix_len = min(s1.length, s3_prefix_len - j);
    k = s1_prefix_len;
    //比较s3[i - 1]与s1[k - 1]相同
    if(s1[k - 1] == s3[i - 1]){
        d[j][k] = d[j][k] || d[j][k - 1];
    }
}

最终结果是d[s2.length][s1.length]。

完整的代码：

    bool isInterleave(string s1, string s2, string s3) {
        const int len1=s1.size();
        const int len2=s2.size();
        const int len3=s3.size();
        if(len1 == 0) return s2 == s3;
        if(len2 == 0) return s1 == s3;
        if(len1 + len2 != len3) return false;
        vector<vector<int>> d(len2+1,vector<int>(len1+1,0));
        d[0][0] = 1;
        for(int i = 0;i <= len3;++i){
            const int s3_prefix_len = i;
            const int s2_prefix_len = min(s3_prefix_len, len2);
            for(int j = 0;j <= s2_prefix_len;++j){
                const int s1_prefix_len = min(len1, s3_prefix_len - j);
                const int k = s1_prefix_len;
                if(k && i && s1[k - 1] == s3[i - 1]) d[j][k] = d[j][k] || d[j][k - 1];
            }
            const int s1_prefix_len = min(s3_prefix_len, len1);
            for(int j = 0;j <= s1_prefix_len;++j){
                const int s2_prefix_len = min(len2, s3_prefix_len - j);
                const int k = s2_prefix_len;
                if(k && i && s2[k - 1] == s3[i - 1]) d[k][j] = d[k][j] || d[k - 1][j];
            }
        }
        return d[len2][len1];
    }

换一种方式递推

设s3(i)表示串s3的长度为i的前缀。s3(i)的最后一个字符为s3[i - 1]。

s2(j)和s1(k)同理。

在上面的讨论中，先固定s3(i)，在固定s2(j)（或s1(j)），然后固定s1(k)（或s2(k)）。

现在，换一种方式递归，先固定s2(j)，再固定s3(k)，最后固定s1(i)。因为s2(j)和s3(k)固定后，能表达的s1(i)也固定。此时，j+k == i。

正确的递推，也要基于相同的字符比较。

递推思路：

d[j][k]表示s2(j)，s1(k)能否匹配s3(j+k);
for(j <= s2.length){
    for(k <= s1.length){
        i =  j + k;
        if(j == 0 且 k == 0) d[j][k] = 1;
        else if(j == 0 且 k > 0){
            if(s3[i - 1] == s1[k - 1]) d[j][k] = d[j][k] || d[j][k - 1];
        }else if(k == 0 且 j > 0){
            if(s3[i - 1] == s2[j - 1]) d[j][k] = d[j][k] || d[j - 1][k];
        }else{
            if(s3[i - 1] == s1[k - 1]) d[j][k] = d[j][k] || d[j][k - 1];
            if(s3[i - 1] == s2[j - 1]) d[j][k] = d[j][k] || d[j - 1][k];
        }
    }
}

结果是d[s2.length][s1.length]。

完整的代码：

    bool isInterleave(string s1, string s2, string s3) {
        const int len1=s1.size();
        const int len2=s2.size();
        const int len3=s3.size();
        if(len1 == 0) return s2 == s3;
        if(len2 == 0) return s1 == s3;
        if(len1 + len2 != len3) return false;
        vector<vector<int>> d(len2+1,vector<int>(len1+1,0));
        for(int j=0;j<=len2;++j){
            for(int k=0;k<=len1;++k){
                const int i = j + k;
                if(j == 0 && k == 0){
                    d[j][k] = 1;
                }else if(j == 0 && k > 0){
                    if(s3[i - 1] == s1[k - 1]) d[j][k] = d[j][k] || d[j][k - 1];
                }else if(k == 0 && j > 0){
                    if(s3[i - 1] == s2[j - 1]) d[j][k] = d[j][k] || d[j - 1][k];
                }else{
                    if(s3[i - 1] == s1[k - 1]) d[j][k] = d[j][k] || d[j][k - 1];
                    if(s3[i - 1] == s2[j - 1]) d[j][k] = d[j][k] || d[j - 1][k];
                }
            }
        }
        return d[len2][len1];
    }

降到一维滚动数组

上面的解法中，j从0到最大值，k也是从0到最大值。说明，计算是从上到下，从左向右。

每个值d[j][k]依赖于d[j-1][k]和d[j][k-1]。而按照计算顺序，d[j-1][k] 先算完， 之后d[j][k-1] ，最后d[j][k]。

容易发现，表格d只需两行即可完成算法。

再进一步，在算完d[j][k]后，d[j-1][k]不会再用于其它处计算。因此，表格d只需一行，简化成滚动数组。

稍作调整，思路：

d[k]表示s2(j)，s1(k)能否匹配s3(j+k);
for(j <= s2.length){
    for(k <= s1.length){
        i =  j + k;
        if(j == 0 且 k == 0) d[k] = 1;
        else if(j == 0 且 k > 0){
            d[k] = d[k - 1] 且 (s3[i - 1] == s1[k - 1]);
        }else if(k == 0 且 j > 0){
            d[k] = d[k] 且 (s3[i - 1] == s2[j - 1]);
        }else{
            d[k] = d[k - 1] 且 (s3[i - 1] == s1[k - 1]) 或
                   d[k] 且 (s3[i - 1] == s2[j - 1]);
        }
    }
}

结果是d[s1.length]。

完整代码：

    bool isInterleave(string s1, string s2, string s3) {
        const int len1=s1.size();
        const int len2=s2.size();
        const int len3=s3.size();
        if(len1 == 0) return s2 == s3;
        if(len2 == 0) return s1 == s3;
        if(len1 + len2 != len3) return false;
        vector<int> d(len1+1,0);
        for(int j=0;j<=len2;++j){
            for(int k=0;k<=len1;++k){
                const int i = j + k;
                if(j == 0 && k == 0){
                    d[k] = 1;
                }else if(j == 0 && k > 0){
                    d[k] = d[k - 1] && (s3[i - 1] == s1[k - 1]);
                }else if(k == 0 && j > 0){
                    d[k] = d[k] && s3[i - 1] == s2[j - 1];
                }else{
                    d[k] = d[k - 1] && (s3[i - 1] == s1[k - 1]) || 
                           d[k] && (s3[i - 1] == s2[j - 1]);
                }
            }
        }
        return d[len1];
    }

解法三 广度优先搜索

设s3(i)表示串s3的长度为i的前缀。s3(i)的最后一个字符为s3[i - 1]。

s2(j)和s1(k)同理。

假设，现在已经确定 s2(j)和s1(k) 能匹配s3(i)，i=j+k。

在这个前提下，确定下一步，

s2(j+1)和s1(k)能否匹配s3(i+1)？

还是 s2(j)和s1(k+1)能否匹配s3(i+1)？

借助上面的动态规划解法，可以将这个问题，转化成从位置(j,k)向(j+1,k)走，还是向(j,k+1)走？

这将问题转成二维隐式无向图上的最短路径问题。用bfs可解此最短路径问题。

当从节点(j,k)能到达节点(s2.length,s1.length)时，表示问题有解。

bfs思路：

Q:位置队列;
vis[j][k]:表示位置(j,k)是否已经到达过;
(0,0)入Q;
vis[0][0]=1;
while(Q中有位置){
    (j,k) = Q出队列;
    if((j,k) == (s2.length,s1.length)) 返回匹配成功;
    //向位置(j+1，k)走?
    if(j < len2 且 s2[j] == s3[j+k] 且 vis[j+1][k] == 0){
        vis[j+1][k] = 1;
        位置(j+1,k)入Q;
    }
    //向位置(1，k+1)走?
    if(k < len1 且 s1[k] == s3[j+k] 且 vis[j][k+1] == 0){
        vis[j][k+1] = 1;
        位置(j,k+1)入Q;
    }
}

完整的代码：

    typedef pair<int,int> Pair;
    bool isInterleave(string s1, string s2, string s3) {
        const int len1=s1.size();
        const int len2=s2.size();
        const int len3=s3.size();
        if(len1 == 0) return s2 == s3;
        if(len2 == 0) return s1 == s3;
        if(len1 + len2 != len3) return false;
        queue<Pair> Q;
        Q.push(make_pair(0,0));
        unordered_map<int,unordered_map<int,int>> vis;
        vis[0][0] = 1;
        while(Q.size()){
            Pair cur=Q.front();Q.pop();
            int _j = cur.first;
            int _k = cur.second;
            if(_j == len2 && _k == len1){
                return true;
            }
            if(_j < len2 && s2[_j] == s3[_j+_k]){
                if(!vis[_j+1][_k]){
                    vis[_j+1][_k] = 1;
                    Q.push(make_pair(_j+1,_k));
                }
            }
            if(_k < len1 && s1[_k] == s3[_j+_k]){
                if(!vis[_j][_k+1]){
                    vis[_j][_k+1] = 1;
                    Q.push(make_pair(_j,_k+1));
                }
            }            
        }
        return false;
    }
