movie recommendation 电影推荐

Posted on 2019年7月30日2019年8月30日by jasonpeng

leetcode 数据库题目全部题解

计算每对电影之间的关联度。用户看了电影A，推荐给他，与电影A关联度最高的其它电影。

这样的关联度如何而来？

从用户给电影的评分记录中，计算出关联度。用户给电影的评论数据：用户 电影 评分。

算法分三个阶段。

阶段一：

统计每部电影的评分总人数。

map.py

输入格式：用户 电影 评分

输出格式：电影 用户 评分

#!/usr/bin/python
#coding=utf-8
import sys

for line in sys.stdin:
    line = line.strip()
    segs = line.split()
    user = segs[0]
    movie = segs[1]
    rating = segs[2]
    print('%s\t%s\t%s'%(movie,user,rating))

reduce.py

输入格式： 电影 用户 评分

输出格式： 用户 电影 评分 电影的评分总人数

#!/usr/bin/python
#coding=utf-8
import sys

last_movie = ''
last_user_rate = []

def emit(movie,user_rate):
    '''
    加上每部电影的评分。
    '''
    for x in user_rate:
        print('%s\t%s\t%s\t%d'%(x[0],movie,x[1],len(user_rate)))

for line in sys.stdin:
    line = line.strip()
    segs= line.split()
    movie = segs[0]
    user_rate = segs[1:]
    if movie != last_movie:
        if last_movie != '':
            emit(last_movie,last_user_rate)
            last_movie = ''
            last_user_rate = []
    last_movie = movie
    last_user_rate.append(user_rate)

if last_movie != '':
    emit(last_movie,last_user_rate)

reduce2.py

将每个用户评论过的电影收集起来。这些电影之间两两组合，统计这些组合的一些参数。

输入格式：用户 <电影1 评分 评分总人数>, <电影2 评分 评分总人数>,......

输出格式：

电影1,电影2 评分1,评分人数1, 评分2,评分人数2, 评分1 x 评分2, 评分1的平方, 评分2的平方

电影2,电影3 评分1,评分人数1, 评分2,评分人数2, 评分1 x 评分2 ,评分1的平方,评分2的平方

#!/usr/bin/python
#coding=utf-8
import sys

last_user = ''
last_mrn = []

def gen(user,mrn):
    '''
    生成输出键
    '''
    #print(mrn)
    mrn.sort(key=lambda t:t[0])
    #print(mrn)
    for i,x in enumerate(mrn):
        for y in mrn[i+1:]:
            ratingProduct = long(x[1]) * long(y[1])
            rating1Squared = long(x[1]) * long(x[1])
            rating2Squared = long(y[1]) * long(y[1])
            print('%s,%s\t%s\t%s\t%s\t%s\t%d\t%d\t%d' % (x[0],y[0],x[1],x[2],y[1],y[2],ratingProduct,rating1Squared,rating2Squared))
    

for line in sys.stdin:
    line = line.strip()
    segs = line.split()
    user = segs[0]
    mrn = segs[1:]
    if user != last_user:
        if last_user != '':
            gen(last_user,last_mrn) 
            last_user = ''
            last_mrn = []
    last_user = user
    last_mrn.append(mrn)

if last_user != '':
    gen(last_user,last_mrn)

reduce3.py

将同一电影对，不同用户的评分综合并计算关联度。

#!/usr/bin/python
#coding=utf-8
import sys
import math
from math import sqrt

last_movie = ''
last_value = []

def pearson_cor(size,dot_product,rating_1sum,rating_2sum,rating_1normsq,rating_2normsq):
    '''
    皮尔逊相关系数
    '''
    numerator = size * dot_product - rating_1sum * rating_2sum
    denominator = sqrt(size * rating_1normsq - rating_1sum * rating_1sum)*sqrt(size * rating_2normsq - rating_2sum * rating_2sum)
    if denominator == 0.0:
        return float('nan')
    return numerator / denominator

def cosine_cor(dot_product,rating_1norm,rating_2norm):
    '''
    余弦相关系数
    '''
    return dot_product / float(rating_1norm * rating_2norm)

def jaccard_cor(in_common,total_a,total_b):
    '''
    Jaccard相关系数
    '''
    return in_common / float(total_a + total_b - in_common)

def gen(movie,value):
    '''
    '''
    group_size = len(value)
    dot_product = 0
    rating_1sum = 0
    rating_2sum = 0
    rating_1normsq = 0
    rating_2normsq = 0
    maxnumofumraters1 = 0
    maxnumofumraters2 = 0
    for x in value:
        dot_product += long(x[4])
        rating_1sum += long(x[0])
        rating_2sum += long(x[2])
        rating_1normsq += long(x[5])
        rating_2normsq += long(x[6])
        maxnumofumraters1 = max(maxnumofumraters1,long(x[1]))
        maxnumofumraters2 = max(maxnumofumraters2,long(x[2]))
    pearson = pearson_cor(group_size,dot_product,rating_1sum,rating_2sum,rating_1normsq,rating_2normsq)
    cosine = cosine_cor(dot_product,sqrt(rating_1normsq),sqrt(rating_2normsq))
    jaccard = jaccard_cor(group_size,maxnumofumraters1,maxnumofumraters2)
    print('%s\t%.2f\t%.2f\t%.2f'%(movie,pearson,cosine,jaccard))

for line in sys.stdin:
    line = line.strip()
    #print(line)
    segs = line.split()
    movie = segs[0]
    value = segs[1:]
    if last_movie != movie:
        if last_movie != '':
            gen(last_movie,last_value)
            last_movie = ''
            last_value = []
    last_movie = movie
    last_value.append(value)

if last_movie != '':
    gen(last_movie,last_value)
