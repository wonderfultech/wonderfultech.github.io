leetcode20 有效的括号 Valid Parentheses

Posted on 2019年4月24日2019年12月4日by jasonpeng

leetcode 数据库题目全部题解

class Solution {
public:
    bool isValid(string s) {
        int hash[256];
        hash[')']='(';
        hash[']']='[';
        hash['}']='{';
        stack<int> S;
        for(int i=0;i<s.size();++i){
            if(s[i]=='(' || s[i]=='[' || s[i]=='{'){
                S.push(s[i]);
            }else if(S.size() && S.top()==hash[s[i]]) S.pop();
            else return false;
        }
        return S.empty();
    }
};
