leetcode60 第k个排列 Permutation Sequence

  • 方法1，dfs+倍数法
  • 方法2，分组法

方法1，dfs+倍数法
当k==1时，剩余数字按递增顺序排
当k==0或k==位置全排列个数时。剩余数字按递减顺序排
找到sub，使sub的阶乘值是小于等于k的最大值
确定位置n-sub处的数字
确定第一部分。位置n-sub前的数字。剩余数字按递增顺序排
确定第二部分。n-sub位置处的数字，注意是第mm个数字。m=k/jie[sub],mm=m+(k%jie[sub] ? 1 : 0);

class Solution {
public:
    const int jie[12]={1,1,2,6,24,120,720,5040,40320,362880,3628800,39916800};
    string getPermutation(int n, int k) {
        vector<int> vis(n,0);
        vector<int> ans(n);
        dfs(n,k,vis,ans,0);
        string r;
        for(int i=0;i<n;++i){
            r+=to_string(ans[i]);
        }
        return r;
    }
    
    void dfs(int n,int k,
             vector<int>& vis,
             vector<int>& ans,int p){
//        if(p>=n) return ;
        //printf("%d %d %d \n",n,k,p);
        const int left=n-p;
        if(k==1){//剩余数字按递增顺序排
            for(int i=0;i<n;++i){
                if(!vis[i]) ans[p++]=i+1;
            }
            return ;
        }else if(!k || k==jie[left]){//剩余数字按递减顺序排
            for(int i=n-1;i>=0;--i){
                if(!vis[i]) ans[p++]=i+1;
            }
            return ;
        }
        int sub=lower_bound(jie,jie+12,k)-jie;
        if(jie[sub]>k) --sub;
//        printf("sub %d jie[%d]:%d \n",sub,sub,jie[sub]);
        //确定位置n-sub处的数字
        //确定第一部分。位置n-sub前的数字
        for(int i=0;i<n && (p<n-(sub+1));++i){
            if(!vis[i]){
                vis[i]=1;
                ans[p++]=i+1;
            } 
        }
        //确定第二部分。n-sub位置处的数字，注意是第mm个数字
        int m=k/jie[sub];
        int mm=m+(k%jie[sub] ? 1 : 0);
        for(int i=0;i<n && mm>0;++i){
            if(!vis[i]){
                if(!--mm){
                    vis[i]=1;
                    ans[p++]=i+1;
                }
            }
        }
        dfs(n,k-m*jie[sub],vis,ans,p);
    }
};

用set替换vis数组。

class Solution {
public:
    const int jie[12]={1,1,2,6,24,120,720,5040,40320,362880,3628800,39916800};
    string getPermutation(int n, int k) {
        vector<int> path(n);
        set<int> digit;
        for(int i=0;i<n;++i){
            digit.insert(i+1);
        }
        dfs(n,k,path,0,digit);
        string ans;
        for(int i=0;i<n;++i){
            ans+=to_string(path[i]);
        }
        return ans;
    }
    
    void dfs(int n,int k,
             vector<int>& path,
             int cur,
             set<int>& digit
            ){
        if(k==1){//剩余数字递增
            for(set<int>::iterator it=digit.begin();it!=digit.end();++it){
                path[cur++]=*it;
            }
            return ;
        }else if(!k || k==jie[n-cur]){
            for(set<int>::reverse_iterator it=digit.rbegin();it!=digit.rend();++it){
                path[cur++]=*it;
            }
            return ;
        }
        //定位k的区间
        int p=lower_bound(jie,jie+12,k)-jie;
        if(jie[p]>k) --p;
        //确定第一部分。填充位置n-p-1之前的位置
        for(set<int>::iterator it=digit.begin();cur<(n-p-1) && it!=digit.end();){
            path[cur++]=*it;
            it=digit.erase(it);
        }
        //确定第二部分。n-p-1位置的值
        int m=k/jie[p];
        int r=k%jie[p];
        int rr=m+(r ? 1 : 0);
        for(set<int>::iterator it=digit.begin();rr>0 && it!=digit.end();){
            if(!--rr){
                path[cur++]=*it;
                it=digit.erase(it);
            }else ++it;
        }
        //递归
        dfs(n,r,path,cur,digit);
    }
};

用数字串替换上面的vis和set，代码更简洁。

class Solution {
public:
    const int jie[12]={1,1,2,6,24,120,720,5040,40320,362880,3628800,39916800};
    string getPermutation(int n, int k) {
        vector<char> path(n);
        string digit("123456789");
        dfs(n,k,path,0,digit);
        string ans;
        for(int i=0;i<n;++i){
            ans+=path[i];
        }
        return ans;
    }
    
    void dfs(int n,int k,
             vector<char>& path,
             int cur,
             string& digit
            ){
        if(k==1){//剩余数字递增
            int i=0;
            while(cur<n){
                path[cur++]=digit[i++];
            }
            return ;
        }else if(!k || k==jie[n-cur]){
            int i=n-cur;
            while(cur<n){
                path[cur++]=digit[--i];
            }
            return ;
        }
        //定位k的区间
        int p=lower_bound(jie,jie+12,k)-jie;
        if(jie[p]>k) --p;
        //确定第一部分。填充位置n-p-1之前的位置
        while(cur<(n-p-1)){
            path[cur++]=*digit.begin();
            digit.erase(digit.begin());
        }
        //确定第二部分。n-p-1位置的值
        int m=k/jie[p];
        int r=k%jie[p];
        int rr=m+(r ? 1 : 0);
        path[cur++]=*(digit.begin()+rr-1);
        digit.erase(digit.begin()+rr-1);
        //递归
        dfs(n,r,path,cur,digit);
    }
};

方法2，分组法
先--k，换成下标
数字个数从n到1迭代
i-1个数字的全排列作为一组，确定k在i个数字的全排列中的哪一组j。注意从n-1开始。
更新k为i-1个数字的全排列。
剩余数字中，下标j是位置n-i上的数字。
删掉已经选出的数字。

class Solution {
public:
    const int jie[12]={1,1,2,6,24,120,720,5040,40320,362880,3628800,39916800};
    string getPermutation(int n, int k) {
        //对下标处理
        --k;
        string ans;
        string nums("123456789");
        for(int i=n;i>=1;--i){
            //i-1个数字的全排列作为一组，确定k在i个数字的全排列中的哪一组j。注意从n-1开始。
            int j=k/jie[i-1];
            //printf("j %d k %d jie %d newk %d \n",j,k,jie[i-1],k%jie[i-1]);
            //更新k为i-1个数字的全排列
            k%=jie[i-1];
            //剩余数字中，下标j是位置n-i上的数字
            ans.push_back(nums[j]);
            //删掉已经选出的数字
            nums.erase(nums.begin()+j);
        }
        return ans;
    }
};

调整数字串的个数。再一次简化代码。

class Solution {
public:
    const int jie[12]={1,1,2,6,24,120,720,5040,40320,362880,3628800,39916800};
    string getPermutation(int n, int k) {
        string d;
        for(int i=0;i<n;++i){
            d+=i+1+'0';
        }
        string ans;
        dfs(n,k,0,d,ans);
        return ans;
    }
    
    void dfs(int n,
             int k,
             int p,
             string& digit,
             string& ans
            ){
        if(k==1){
            ans+=digit;
            return;
        }else if(k==0 || k==jie[n-p]){
            reverse(digit.begin(),digit.end());
            ans+=digit;
            return ;
        }
        
        int left=lower_bound(jie,jie+12,k)-jie;
        if(jie[left]>k) --left;
        //确定位置n-left-1之前的
        for(int i=p;i<(n-left-1);++i){
            ans+=digit[0];
            digit.erase(digit.begin());
        }
        //确定n-left-1处的字符
        int g=k/jie[left];
        int r=k%jie[left];
        //注意此处是序号。第rr个数字，减1后是下标
        int rr=g+(r ? 1 : 0)-1;
        ans+=digit[rr];
        digit.erase(digit.begin()+rr);
        dfs(n,r,ans.size(),digit,ans);
    }
};
