leetcode102 二叉树的层次遍历 Binary Tree Level Order Traversal

Posted on 2019年4月17日2019年12月4日by jasonpeng

leetcode 数据库题目全部题解

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> ans;
        queue<TreeNode*> Q;
        if(!root) return ans;
        Q.push(root);
        while(Q.size()){
            int cnt=Q.size();
            ans.push_back(vector<int>());
            while(cnt--){
                TreeNode* p=Q.front();Q.pop();
                ans.back().push_back(p->val);
                if(p->left) Q.push(p->left);
                if(p->right)Q.push(p->right);
            }
        }
        return ans;
    }
};
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> ans;
        queue<TreeNode*> Q;
        if(!root) return ans;
        Q.push(root);
        TreeNode* right=root,*nextright=NULL;
        ans.push_back(vector<int>());
        while(Q.size()){
            TreeNode* p=Q.front();Q.pop();
            ans.back().push_back(p->val);
            if(p->left) Q.push(nextright=p->left);
            if(p->right)Q.push(nextright=p->right);
            if(p==right){
                ans.push_back(vector<int>());
                right=nextright;
            }
        }
        while(ans.size() && ans.back().empty()) ans.pop_back();
        return ans;
    }
};
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> ans;
        dfs(root,0,ans);
        return ans;
    }
    
    void dfs(TreeNode* rt,int level,
             vector<vector<int>>& ans){
        if(!rt) return ;
        if(level==ans.size()) ans.push_back(vector<int>());
        ans[level].push_back(rt->val);
        dfs(rt->left,level+1,ans);
        dfs(rt->right,level+1,ans);
    }
};
