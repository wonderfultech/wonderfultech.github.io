leetcode445 两数相加 II Add Two Numbers II

Posted on 2019年10月18日2019年12月4日by jasonpeng

解法一 双栈算法

思路：遍历两个链表，将指向节点的指针存入两个栈中。从栈顶开始取节点的值求和。每求出一对节点的和，出栈节点。直到两栈都为空。

例如：(7 -> 2 -> 4 -> 3) + (5 -> 6 -> 4)

遍历链表得出两个栈。A=[7,2,4,3],B=[5,6,4]。

栈顶求和3+4=7。栈顶弹出。A=[7,2,4],B=[5,6]。

栈顶求和4+6=10。值为0，进1。栈顶弹出。A=[7,2],B=[5]。

栈顶求和2+5+1=8。栈顶弹出。A=[7],B=[]。

栈顶求和7+0=7。栈顶弹出。A=[],B=[]。

如此，结果是7,8,0,7。

思路：

s1，s2:两个栈;
for(ListNode x in l1){
    x 入 s1;
}
for(ListNode x in l2){
    x 入 s2;
}
d = 0;
while(s1不空 || s2不空){
    if(s1不空) d+=s1栈顶出栈;
    if(s2不空) d+=s2栈顶出栈;
    值d % 10，挂入结果链表;
    进位 d /=10;
}
if(d){
   d，挂入结果链表;
}

完整代码：

    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        vector<ListNode*> l1v,l2v;
        to_vector(l1,l1v);
        to_vector(l2,l2v);
        int i=l1v.size()-1, j=l2v.size()-1;
        int d = 0;
        ListNode *head = NULL;
        while(i >= 0 || j >= 0){
            if(i >= 0) d += l1v[i--]->val;
            if(j >= 0) d += l2v[j--]->val;
            ListNode* p = new ListNode(d%10);
            p->next = head;
            head = p;
            d /= 10;
        }
        if(d) {
            ListNode* p = new ListNode(d);
            p->next = head;
            head = p;
        }
        return head;
    }
    
    void to_vector(ListNode* a,vector<ListNode*>& v){
        while(a){
            v.push_back(a);
            a = a->next;
        }
    }

解法二 递归法

思路：计算两个链表的长度。在长度短的链表前补上值为0的节点。再以相等长度的链表，递归求和。结束时，删除补上的节点。

例如：(7 -> 2 -> 4 -> 3) + (5 -> 6 -> 4)

先在 (5 -> 6 -> 4) 前补上0，得出(0->5->6->4)。

递归求和时。最右边的节点先算，并将进位递归返回。

递归的思路：


dfs(A,B){
    if(A为空 且 B为空) 返回0;
    尾部求和进位 r = dfs(A的下一个节点，B的下一个节点);
    求和值 d = A.val + B.val + r;
    新节点值 v = d % 10;
    返回 进位 d/10;
}

求链表长度代码：

    int getlen(ListNode* A){
        int c = 0;
        while(A){
            ++c;
            A = A->next;
        }
        return c;
    }

完整代码：

    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        int len1 = getlen(l1);
        int len2 = getlen(l2);
        bool ok = (len1 >= len2);
        const int diff = ( ok ? len1-len2 : len2-len1);
        ListNode *H=NULL,*T=NULL;
        //补上值为0的节点
        for(int i = 0;i< diff;++i){
            ListNode* p = new ListNode(0);
            if(!H) T=p;
            p->next = H;
            H = p;
        }
        ListNode* ans = NULL;
        if(diff){
            if(ok){
                T->next = l2;
            }else{
                T->next = l1;
            }

            int r = dfs(H,ok ? l1 : l2,ans);
            if(r){
                ListNode* p = new ListNode(r);
                p->next = ans;
                ans = p;    
            }
            T->next = NULL;
            //删除补上的节点
            while(H){
                ListNode* p=H->next;
                delete H;
                H = p;
            }
        }else{
            int r = dfs(l1,l2,ans);
            if(r){
                ListNode* p = new ListNode(r);
                p->next = ans;
                ans = p;    
            }
        }     
        
        return ans;
    }
    
    int dfs(ListNode* A,ListNode* B,ListNode* & ans){
        if(!A && !B) return 0;
        int r = dfs(A->next,B->next,ans)+A->val+B->val;
        ListNode* p = new ListNode(r%10);
        p->next = ans;
        ans = p;
        return r/10;
    }
