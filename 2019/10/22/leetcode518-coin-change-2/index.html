解法一 回溯法（超时）

思路：面值为S，硬币coins。枚举每种硬币组合。统计组合的个数。

为了去除重复组合，要求组合中的硬币按递增的顺序排列。

例如：S=3 coins=[1,2]

[1,2]和[2,1]是同一种。用递增就限制了[2,1]的出现。

在面值S时，选中coins[i]。下一步，要在面值S-coins[i]时，选中coins[j] 且 coins[j] <= S-coins[i] 且 coins[j] >= coins[i] 。

这样，面值逐步减小直到0或非0。如果面 值 能减到0，算一种组合。如果面值不能减到0，就不算一种组合。

思路：

choice[i]表示每个位置选中的硬币面值
dfs(S,cur,choice){
    if(S == 0) 返回 1;
    if(S < 0) 返回 0;
    res = 0;
    for(c in coins){
        if(c <= S){
            if(cur == 0 || choice[cur-1] <= c){
                choice[cur]=c;
                r = dfs(S-c,cur+1,choice);
                res += r;
            }
        }
    }
    返回 res;
}

完整代码：

    int change(int amount, vector<int>& coins) {
        sort(coins.begin(),coins.end());
        vector<int> path(amount+1,0);
        return dfs(coins,path,0,amount);
    }
    
    int dfs(const vector<int>& coins,
            vector<int>& path,
            int cur,
            int S){
        if(!S) return 1;
        if(S<0) return 0;
        int res = 0;
        for(int i=0;i<coins.size();++i){
            if(coins[i] <= S){
                if(!cur || path[cur-1] <= coins[i]){
                    path[cur] = coins[i];
                    int r = dfs(coins,path,cur+1,S-coins[i]);
                    res += r;
                }
            }else break;
        }
        return res;
    }

解法二 回溯法

在解法一的基础上，考虑减枝。

解法一的关键点：

在面值S时，选中coins[i]。下一步，要在面值S-coins[i]时，选中coins[j] 且 coins[j] <= S-coins[i] 且 coins[j] >= coins[i] 。

那么，如果S-coins[i]的组合数已经算出，对S-coins就无需再递归。

设d[S]表示面值为S时的组合数。

现在考虑由d[S-coins[i]]能否算出d[S]。

例如：考虑面值S=1，2，3，4，5，6时，coins=[1,2,5]的组合情形

1=1+0

2=1+1
2=2+0

3=1+1+1
3=1+2

4=1+1+1+1
4=1+1+2
4=2+2

5=1+1+1+1+1
5=1+1+1+2
5=1+2+2
5=5+0

6=1+1+1+1+1+1
6=1+1+1+1+2
6=1+1+2+2
6=1+5
6=2+2+2

d[1]=1,d[2]=2

现在，面值S=3，考虑d[3]，

选中coins[0]=1，那么剩余面值S-coins[0]=2。而d[2]=2。d[3]=2。

选中coins[1]=2，那么剩余面值S-coins[1]=1。而d[1]=1。d[3]=2+1=3。

得出d[3]=3。而实际d[3]=2。错误！

因为选中coins[1]=2后，剩余面值1。此时3=2+1.

而选中coins[0]=1时，3=1+1+1 ，3=1+2。显然有重复。

这说明d[S]的定义无法排除重复组合。需要扩充定义。

设d[S][j]表示面值为S，组合的第一个硬币是coins[j]时的组合种数。还是要求组合中，硬币的面值递增。

再看上面的例子。

d[1][0]=1

d[2][0]=1,d[2][1]=1

那么d[3][0]，剩余面值3-1=2。面值2时，选择第一个硬币是coins[0]=1。d[2][0]=1。 选择第一个硬币是coins[1]=2。d[2][1]=1。 d[3][0]=d[2][0]+d[2][1]=2。

d[3][1]， 剩余面值3-2=1。面值1小于面值2，这种组合被排除。 d[3][1] =0。

面值S=3的总组合数=d[3][0]+d[3][1]=2+0=2.

再看S=4.

d[4][0]，剩余面值4-1=3.

面值3时，选择第一个硬币是coins[0]=1。d[3][0]=2。 选择第一个硬币是coins[1]=2。d[3][1]=0。 d[4][0]=d[3][0]+d[3][1]=2。

d[4][1], 剩余面值4-2=2.

面值2时，选择第一个硬币是coins[1]=2。d[2][1]=1。 d[4][1]=1.

面值S=4的总组合数=d[4][0]+d[4][1]=2+1=3.

可以看出,重新定义的d[i][j]能排除掉重复组合数.

完整思路: 基于解法一扩充.

d[S][i]:表示面值为S，组合的第一个硬币是coins[i]时的组合种数。还是要求组合中，硬币的面值递增。
d[S][i]=-1;初始值都为-1.
choice[i]表示每个位置选中的硬币面值
dfs(S,cur,choice,d){
    if(S == 0) 返回 1;
    if(S < 0) 返回 0;
    res = 0;
    for(i,c in coins){
        if(c <= S){
            if(cur == 0 || choice[cur-1] <= c){
                if(d[S][i] 不是 -1){
                    res += d[S][i];
                }else{
                    d[S][i] = 0;
                    choice[cur]=c;
                    r = dfs(S-c,cur+1,choice,d);
                    d[S][i] += r;
                    res += r;
                }
            }
        }
    }
    返回 res;
}

完整代码:

    int change(int amount, vector<int>& coins) {
        sort(coins.begin(),coins.end());
        vector<vector<int>> d(amount+1,vector<int>(coins.size()+1,-1));
        vector<int> path(amount+1,0);
        return dfs(coins,path,0,amount,d);
    }
    
    int dfs(const vector<int>& coins,
            vector<int>& path,
            int cur,
            int S,
            vector<vector<int>>& d){
        vector<int>& ans = d[S];
        if(!S) return 1;
        int res = 0;
        for(int i=0;i<coins.size();++i){
            if(coins[i] <= S){
                if(!cur || path[cur-1] <= coins[i]){
                    if(ans[i] > -1){
                        res += ans[i];
                    }else{
                        ans[i] = 0;
                        path[cur] = coins[i];
                        int r = dfs(coins,path,cur+1,S-coins[i],d);
                        ans[i] += r;
                        res += r;                        
                    }
                }
            }else break;
        }
        return res;
    }

解法三 动态规划

将解法二的递归算法换成递推算法.

设d[S][j]表示面值为S，组合的第一个硬币是coins[j]时的组合种数。还是要求组合中，硬币的面值递增。

从解法二的递归算法中,可以发现:

当S-coins[j] == 0时,d[S][j] += 1.

当S-coins[j] > 0时, 对每个k, k满足条件coins[j] <= coins[k] <= S - coins[j], 有d[S][j] += d[S-coins[j]][k]. 意思是第一个硬币面值是coins[j],第二个硬币面值是coins[k].

完整思路:

for(S in [0,...,amount]){
    for(j in [0,...,coins.length-1]){
        if(coins[j] <= S){
            r = S-coins[j];
            if(r==0) d[S][j] += 1;
            else for(k 满足 coins[j] <= coins[k] <= r){
                d[S][j] += d[r][k];
            }
        }
    }
}
结果 = d[S][0,...,coins.length-1]的总和;

完整代码:

    int change(int amount, vector<int>& coins) {
        sort(coins.begin(),coins.end());
        if(!amount) return 1;
        vector<vector<int>> d(amount+1,vector<int>(coins.size()+1,0));
        
        for(int i=0;i<=amount;++i){
            vector<int>& ans=d[i];
            int res = 0;
            for(int j=0;j<coins.size();++j){
                if(coins[j] <= i){
                    int rest = i-coins[j];
                    if(!rest){
                        ans[j] += 1;
                    }else
                    for(int k=j;k < coins.size() && coins[k] <= rest;++k){
                        ans[j] += d[rest][k];
                    }
                        
                    res += ans[j];
                }else break;
            }
            if(i == amount){
                ans[coins.size()] = res;
            }
        }
        return d[amount][coins.size()];
    }

解法四 回溯法

在前三个算法中,以面值S为主. 确定一个面值S后, 求每个S的组合数. 增加S直到完成.

现在转换思路,先确定硬币,然后找出硬币组成面值S的组合数.

例如: amount = 5, coins = [1, 2, 5]

不用硬币时, 只能组成面值0. 面值0的组合数是1,其它都是0.

面值 0  1  2  3  4  5
不用 1  0  0  0  0  0

只用硬币coins[0]=1时,可以组成1到5的面值;

不用0,可以组成面值0.其它全是1,就能组成相应的面值.

1=1+0
2=1+1
3=1+1+1
4=1+1+1+1
5=1+1+1+1+1
面值 0  1  2  3  4  5
不用 1  0  0  0  0  0
用0  1  1  1  1  1  1

用硬币coins[0]=1和coins[1]=2时,可以组成哪些面值了.

显然1能表示的哪些也包括在内.

加上2后有哪些?

1=1+0

2=1+1
2=2+0

3=1+1+1
3=1+2

4=1+1+1+1
4=1+1+2
4=2+2

5=1+1+1+1+1
5=1+1+1+2
5=1+2+2

面值2,不用coins[1]时,已有1种组合. 用1次coins[1]时,还剩面值0.面值0有1种组合.

所以面值2有2种组合.

面值4,不用coins[1]时,已有1种组合.

用1次coins[1],还剩面值2. 面值2,在不用coins[1]时有1种组合.

用2次coins[1],还剩面值0. 面值0,在不用coins[1]时有1种组合.

所以,面值4有3种.

面值5,不用coins[1]时,已有1种组合.

用1次coins[1],还剩面值3. 面值3,在不用coins[1]时有1种组合.

用2次coins[1],还剩面值1. 面值1,在不用coins[1]时有1种组合.

所以,面值5有3种.

面值 0  1  2  3  4  5
不用 1  0  0  0  0  0
用0  1  1  1  1  1  1  
用1  1  1  2  2  3  3

用硬币coins=[1,2,5]时,可以组成哪些面值了.

用coins[2]=5,能起作用的是面值5.

面值5,不用coins[2]时,已有3种组合.

用1次coins[2],还剩面值0. 面值0,在不用coins[2]时有1种组合.

所以,面值5有4种.

5=1+1+1+1+1
5=1+1+1+2
5=1+2+2
5=5+0

因此,在已经使用硬币coins[0,...,i]组成面值[0,...,S]的基础上. 新增硬币coins[i+1].

并由硬币coins[0,...,i+1]组成面值[0,...,S].

对任意面值t, 0 <= t <= S.

t的组合数分成两部分.

第一部分是不用硬币coins[i+1]的组合数,即仅用硬币coins[0,...,i].

第二部分是用硬币coins[i+1]共k(1 <= k) 次的情况. 其值 = 面值S-k*coins[i+1] 不用coins[i+1]的组合数 之和.

定义d[i][S]表示 已经使用硬币coins[-1,0,...,i]组成面值S的组合数.

其中-1表示不用硬币。

思路:

dfs(i,S,d){
    if(i == -1 且 S == 0) 返回1;
    if(i == -1) 返回0;
    //第一部分
    d[i][S] += dfs(i-1,S);
    //第二部分
    for(k in [1,...]){
        if(S-k*coins[i] >= 0){
            d[i][S] += dfs(i-1,S-k*coins[i],d);
        }
    }
}

完整代码:

    int change(int amount, vector<int>& coins) {
        vector<vector<int>> d(coins.size()+1,vector<int>(amount+1,-1));
        return dfs(coins.size()-1,amount,coins,d);
    }

    int dfs(int i,int S,const vector<int>& coins,vector<vector<int>>& d){
        if(i == -1 && S==0) return 1;
        if(i == -1) return 0;
        
        int& ans=d[i][S];
        if(ans > -1) return ans;
        ans = 0;
        ans += dfs(i-1,S,coins,d);
        int t;
        for(int k=1;(t=S-k*coins[i]) >= 0;k++){
            int r = dfs(i-1,t,coins,d);
            ans += r;
        }
        return ans;
    }

解法五 动态规划

基于解法四，将递归改为递推。

定义d[i][S]表示 已经使用i个硬币组成面值S的组合数.

根据解法四。

当i == 0时，S ==0时，d[i][S] = 1。 S>0时，d[i][S] = 0。

当i >0 时，分成两部分。不用coins[i-1]时，d[i][S] = d[i-1][S]。

用coins[i-1]共k次时，d[i][S] += d[i-1][S - k * coins[i-1]]。

完整代码：

    int change(int amount, vector<int>& coins) {
        vector<vector<int>> d(coins.size()+1,vector<int>(amount+1,0));
        for(int i=0;i<=coins.size();++i){
            for(int j=0;j<=amount;++j){
                if(i == 0 && j == 0){
                    d[i][j] = 1;
                }else if(i == 0 && j){
                    d[i][j] = 0;
                }else{
                    int t;
                    for(int k=0;(t = j-k*coins[i-1]) >=0;++k){
                        d[i][j] += d[i-1][t];
                    }
                }
            }
        }
        return d[coins.size()][amount];
    }

降到二维滚动数组

从上面的可看到，d[i][S]只依赖d[i-1][X]的内容。用二维数组就可表示。注意，d[i][j]在复用前要先清0.

    int change(int amount, vector<int>& coins) {
        vector<vector<int>> d(coins.size()+1,vector<int>(amount+1,0));
        for(int i=0;i<=coins.size();++i){
            int r = i & 1;
            for(int j=0;j<=amount;++j){
                if(i == 0 && j == 0){
                    d[r][j] = 1;
                }else if(i == 0 && j){
                    d[r][j] = 0;
                }else{
                    int t;
                    d[r][j] = 0;
                    for(int k=0;(t = j-k*coins[i-1]) >=0;++k){
                        d[r][j] += d[r ^ 1][t];
                    }
                }
            }
        }
        return d[coins.size() & 1][amount];
    }

进一步简化

                    for(int k=0;(t = j-k*coins[i-1]) >=0;++k){
                        d[r][j] += d[r ^ 1][t];
                    }

从这个循环中，

可发现上d[r][j-coins[i-1]]在前面的循环中已经算出。

假设其等于，k是0到p时，d[r ^ 1][k]的和。即 d[r][j-coins[i-1]] = d[r^1][j]+d[r^1][j-coins[i-1]]+,...,+ d[r^1][j- p * coins[i-1] ] 。

现在求d[r][j]。

相当于k扩展到了p+1。

只需要的在 d[r][j-coins[i-1]] 基础上加上d[r ^ 1][j]即可。

d[r][j] = d[r ^ 1][j] + d[r][j-coins[i-1]]

    int change(int amount, vector<int>& coins) {
        vector<vector<int>> d(coins.size()+1,vector<int>(amount+1,0));
        for(int i=0;i<=coins.size();++i){
            int r = i & 1;
            for(int j=0;j<=amount;++j){
                if(i == 0 && j == 0){
                    d[r][j] = 1;
                }else if(i == 0 && j){
                    d[r][j] = 0;
                }else{
                    int t;
                    d[r][j] = d[r ^ 1][j];
                    if((t = j-coins[i-1]) >=0){
                        d[r][j] += d[r][t];
                    }
                }
            }
        }
        return d[coins.size() & 1][amount];
    }

降到一维滚动数组

从上面的结果可看出。d[r][j]变得只与d[r ^ 1][j]和d[r][j-coins[i-1]]有关了。

d[r ^ 1][j]是其上一行得值。 d[r][j-coins[i-1]] 在前面的循环中已经算出。

这时，可将d简化到一维数组了。

    int change(int amount, vector<int>& coins) {
        vector<int> d(amount+1,0);
        for(int i=0;i<=coins.size();++i){
            for(int j=0;j<=amount;++j){
                if(i == 0 && j == 0){
                    d[j] = 1;
                }else if(i == 0 && j){
                    d[j] = 0;
                }else{
                    int t;
                    if((t = j-coins[i-1]) >=0){
                        d[j] += d[t];
                    }
                }
            }
        }
        return d[amount];
    }
