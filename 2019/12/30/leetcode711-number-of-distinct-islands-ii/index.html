解法一 深度优先搜索

本题与uva1602有点类似。

思路：用dfs求出连通分量。每个连通分量就是一座岛屿。统计岛屿个数即可。

关键点在于如何判断两个岛屿是否相同。

分成三步完成：

1，表示岛屿。

2，每个岛屿有六种形态：原始， 顺时针旋转 90°，180°，270° , 左右翻转，上下翻转 。平移，旋转，反转操作怎么完成。

3，比较两个岛屿是否相同。只要每个岛屿六个状态中，有一对相同，那么两个岛屿就相同。

岛屿表示

岛屿就是连通分量。用坐标序列表示岛屿比较方便，也便于旋转，翻转状态的计算。为表示 岛屿 的六种状态，用六个坐标序列表示岛屿。

用第一个坐标序列表示元素状态。

node{
    //data[0]表示第一个坐标序列
    data[6]=[[],[],[],[],[],[]];
    add(x,y){
        (x,y)入data[0];
    }
}

旋转,翻转,平移,排序

连通分量用坐标序列表示。

将坐标(x,y)看作向量。利用解析几何的知识。计算旋转，翻转，平移。

向量(x,y)旋转90度和270度: 构成正交向量。向量内积为0。

因此，旋转90度的向量为(y,-x)。270度的向量是(-y,x).

向量(x,y)旋转180度 : 取反向量(-x,-y)即可。

向量(x,y)绕y轴翻转: x取反即可。(-x,y)。

向量(x,y)绕x轴翻转: y取反即可。(x,-y)。

连通分量的旋转和翻转，通过对其中每个点，进行旋转和翻转来完成。

连通分量的平移，是为了让所有的连通分量都集中在坐标原点附近。这样，相同的连通分量，再通过合适的旋转，翻转，就一定会重合。再判断坐标序列是否相等即可了。

将原始状态data[0],旋转，翻转
to_90(v){
    for((x,y) in data[0]){
        (y,-x)入v;
    }
}
to_270(v){
    for((x,y) in data[0]){
        (-y,x)入v;
    }
}
to_180(v){
    for((x,y) in data[0]){
        (-x,-y)入v;
    }
}
X_flip(v){
    for((x,y) in data[0]){
        (x,-y)入v;
    }
}
Y_flip(v){
    for((x,y) in data[0]){
        (-x,y)入v;
    }
}

连通分量的平移，需要一个平移向量。

将连通分量中，x坐标的最小值xmin，和y坐标的最小值ymin，构成的向量(xmin,ymin)当作平移向量。要注意 (xmin,ymin) 不一定在连通分量中。

将连通分量中，所有的点(x,y)减去向量 (xmin,ymin) , 完成连通分量向坐标原点的平移。

先将连通分量平移到坐标原点，再旋转，翻转，还是先将连通分量旋转，翻转，再平移到坐标原点？

在本题中，翻转，旋转都是刚体变换，谁先谁后本质上没有什么区别。

为了不用重复计算平移，选择，将连通分量旋转，翻转，再平移到坐标原点。

当做完，旋转，翻转，平移后，对坐标排序，便于坐标序列的比较。

比较准则，x坐标小的排前面，x坐标相等的，y坐标小的排前面。

align(v){
    xmin=INT_MAX;
    ymin=INT_MAX;
    for((x,y) in v){
        xmin=min(xmin,x);
        ymin=min(ymin,y);
    }
    for((x,y) in v){
        (x,y) = (x-xmin,y-ymin);
    }
    sort(v);//对坐标排序
}
align(){
    to_90(data[1]);
    to_270(data[2]);
    to_180(data[3]);
    X_flip(data[4]);
    Y_flip(data[5]);
    for(i < 6){
        align(data[i]);
    }
}

哈希函数

本题中用哈希表快速判断连通分量是否相等。

为连通分量设计哈希函数，使相同的连通分量，生成相同的哈希值。但，相同的哈希值不见得是由相同的连通分量生成。

取连通分量中所有坐标的x和y的异或值，为哈希值。

hash_code(){
    r=0;
    for(i < 6){
        for((x,y) in data[i]){
            r = r ^ x ^ y;
        }
    }
    返回r;
}

判断连通分量是否相等

有了哈希函数，可以快速区分不同的连通分量。但是，当哈希值相等时，只能比较坐标序列是否相等。

比较准则：

1，原始坐标序列长度不等，一定不相等；

2，两组坐标序列中，只要有一对相等，两个连通分量就相等;

operator==(other){
    if(data[0].length != other.data[0].length) 返回false;
    for(i < 6){
        for(j < 6){
            if(data[i] == other.data[j]) 返回true;
        }
    }
    返回false;
}

dfs求连通分量

用上面的哈希函数和比较函数，构造哈希表hash。

从一个值为1的起点开始，用dfs将其上下左右，凡是1的点都纳入到同一个连通分量。直到，每个点的周围都没有1为止。

每找到一个连通分量，将其插入哈希表hash去重。

最后，hash中连通分量数就是结果。

hash=[];
R,C;行数列数
vis[x][y]=0;节点是否访问过
grid[x][y];图
cnt=0;连通变量编号
for(x < R){
    for(y < C){
        if(grid[x][y]==1 且 vis[x][y]==0){
            ++cnt;
            data=[];
            dfs(grid,vis,x,y,cnt,data);
            node new_node(data);
            new_node.align();//旋转，翻转，平移，排序
            new_node入hash;
        }
    }
}

hash中的连通分量数是结果；

dfs(grid,vis,x,y,c,data){
    vis[x][y]=c;
    (x,y)入data;
    for((nx,ny) in (x,y)的上下左右){
        if((nx,ny)在图中 且 grid[nx][ny]==1 且  vis[nx][ny]==0){
            dfs(grid,vis,nx,ny,c,data);
        }
    }
}

完整代码

    typedef pair<int,int> Pair;
    struct node{
        vector<vector<Pair>> data;
        node():data(vector<vector<Pair>>(6,vector<Pair>())){}
        
        void add(int x,int y){
            data[0].push_back(make_pair(x,y));
        }
        
        void align(vector<Pair>& v)const{
            Pair a={INT_MAX,INT_MAX};
            for(int i=0;i<v.size();++i){
                a.first = min(a.first,v[i].first);
                a.second = min(a.second,v[i].second);
            }
            for(int i=0;i<v.size();++i){
                v[i].first -= a.first;
                v[i].second -= a.second;
            }
            sort(v.begin(),v.end());
        }
        
        void to_90(vector<Pair> & v)const{
            v.clear();
            for(int i=0;i<data[0].size();++i){
                v.push_back(make_pair(data[0][i].second,-data[0][i].first));
            }
        }
        
        void to_180(vector<Pair>& v)const{
            v.clear();
            for(int i=0;i<data[0].size();++i){
                v.push_back(make_pair(-data[0][i].first,-data[0][i].second));
            }
        }
        
        void X_flip(vector<Pair>& v)const{
            v.clear();
            for(int i=0;i<data[0].size();++i){
                v.push_back(make_pair(data[0][i].first,-data[0][i].second));
            }
        }
        
        void Y_flip(vector<Pair>& v)const{
            v.clear();
            for(int i=0;i<data[0].size();++i){
                v.push_back(make_pair(-data[0][i].first,data[0][i].second));
            }
        }
        
        void to_270(vector<Pair>& v)const{
            v.clear();
            for(int i=0;i<data[0].size();++i){
                v.push_back(make_pair(-data[0][i].second,data[0][i].first));
            }
        }
        
        void align(){
            to_90(data[1]);
            to_180(data[2]);
            to_270(data[3]);
            X_flip(data[4]);
            Y_flip(data[5]);
            for(int i=0;i<6;++i){
                align(data[i]);
            }
            
        }
        
        bool operator==(const node& o)const{
            if(data[0].size() != o.data[0].size()) return false;
            for(int i=0;i<6;++i){
                for(int j=0;j<6;++j){
                    if(data[i] == o.data[j]) return true;
                }    
            }
            return false;
        }
        
        size_t hash_code()const{
            size_t r=0;
            for(int j=0;j<6;++j){
                for(int i=0;i<data[j].size();++i){
                    r ^= data[j][i].first ^ data[j][i].second;
                }    
            }       
            return r;            
        }
        
        string to_str()const{
            stringstream ss;
            for(int j=0;j<6;++j){
                for(int i=0;i<data[j].size();++i){
                    ss << " [" << data[j][i].first <<" " << data[j][i].second << " ] " ;
                }
                ss << "\n";    
            }
            return ss.str();
        }
    };
    struct node_hash{
        size_t operator()(const node& o)const{
            return o.hash_code();
        }
    };
    struct node_pred{
        bool operator()(const node& a,const node& b)const{
            return a==b;
        }
    };
    int R,C;
    int numDistinctIslands2(vector<vector<int>>& grid) {
        if(!(R=grid.size()) || !(C=grid[0].size())) return 0;
        vector<vector<int>> vis(R,vector<int>(C,0));
        int cnt=0;
        unordered_set<node,node_hash,node_pred> S;
        for(int i=0;i<R;++i){
            for(int j=0;j<C;++j){
                if(grid[i][j]==1 && vis[i][j] == 0){
                    node nd;
                    ++cnt;
                    dfs(grid,vis,i,j,cnt,nd.data[0]);
                    nd.align();
                    S.insert(nd);
                }
            }
        }
        return S.size();
    }
    
    void dfs(vector<vector<int>>& grid,
             vector<vector<int>>& vis,int x,int y,int c,vector<Pair>& v){
        vis[x][y]=c;
        v.push_back({x,y});
        const static int dx[4]={1,-1,0,0};
        const static int dy[4]={0,0,1,-1};
        for(int i=0;i<4;++i){
            const int nx=x+dx[i];
            const int ny=y+dy[i];
            if(in_grid(nx,ny) && grid[nx][ny]==1 && vis[nx][ny]==0){
                dfs(grid,vis,nx,ny,c,v);
            }
        }
    }
    bool in_grid(int x,int y){
        return x>=0 && x<R && y>=0 && y<C;
    }
