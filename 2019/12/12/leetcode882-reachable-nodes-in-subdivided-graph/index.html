解法一 广度优先搜索算法（超时）

思路：将分割后的边补充到无向图中。再从0结点开始，bfs计算每个节点到节点0的最短距离。统计距离<=M的节点数。

用邻接表G结构表示无向图。G中放置节点编号.

邻接表G<节点i,[节点j,....]>.

0到N-1表示原图中的节点。

构造新无向图

边e(i,j,k)，表示在原来的边(i,j)上插入k个新节点。新节点从N开始计数。分割成k+1个新边。用这k+1个新边代替原来的边(i,j)，并插入原图。

构建新图的思路：

G[i]=[];新图
s=N;//新节点从N开始
for(e in edges){
    u=e[0];
    v=e[1];
    //新增e[2]个节点，原图中插入e[2]个新边
    for(j < e[2]){
        add_edge(G,u,s);//新增边u->s
        u=s++;
    }
    add_edge(G,u,v);
}

bfs寻找距离<=M的节点

为了提高效率，这里用优先级队列Q代替bfs中的普通队列。

离0节点距离越短的节点，优先级越高，优先出队列。

同时设置数组vis[i]，兼顾bfs算法去重，以及记录节点i与节点0的距离。

vis[i]=0;
Q=[];优先级队列;
ans=1;结果
vis[0]=1;
for(j in G[0]){
    if(vis[j] ==0){
        ++vis[j];
        Q.push(<j,vis[j]>);
        ++ans;
    }
}
while(Q不空){
    <x,x_dist>=Q.top出Q;
    if(vis[x] >= M) 退出；最短的距离都超过M了，不用算了;
    for(j in G[x]){
        if(vis[j] ==0){
            ++vis[j];
            Q.push(<j,vis[j]>);
            ++ans;
        }
    }
}

代码：

    struct node{
        int p;
        int d;
        node(int _p = 0,int _d=0):p(_p),d(_d){}
        bool operator<(const node& o)const{
            return d > o.d;
        }
    };
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        if(M ==0 ) return 1;
        vector<vector<int>> grid(N,vector<int>());
        int s=N;
        for(int i=0;i<edges.size();++i){
            vector<int>& e=edges[i];
            const int n=e[2];
            int u=e[0];
            int v=e[1];
            for(int j=0;j<n;++j){
                add_edge(grid,u,s);
                u=s++;
            }
            add_edge(grid,u,v);
        }
        
        vector<int> vis(s,0);
        vis[0] = 1;
        priority_queue<node> Q;
        int ans=1;
        for(int j=0;j<grid[0].size();++j){
            int p=grid[0][j];
            if(vis[p]) continue;
            ++vis[p];
            if(vis[p] < M){
                Q.push(node(p,vis[p]));                
            }

            ++ans;
        }
        while(Q.size()){
            node nd=Q.top();Q.pop();
            if(vis[nd.p] == M) break;
            //if(vis[nd.p] == M) continue;
            for(int j=0;j<grid[nd.p].size();++j){
                int q=grid[nd.p][j];
                if(vis[q]) continue;
                vis[q] = vis[nd.p] + 1;
                if(vis[q] < M){
                    Q.push(node(q,vis[q]));
                }
                ++ans;
            }
        }
        return ans;
    }
    
    void add_edge(vector<vector<int>>& grid,int u,int v){
        grid[u].push_back(v);
        if(v >= grid.size()){
            grid.push_back(vector<int>());
        }
        grid[v].push_back(u);
    }

解法二 最短路径法

解法一中 ， 在M值很大时，会产生大量的新增节点，导致bfs效率低下。

换一种思路，不再原图中插入新增节点。

而是将新增加的边，作为原图中边的距离权重。

用单源最短路径算法，计算从节点0到所有节点i的最短距离vis[i]。

原图中的边e(i,j,w)。

当dist[i] <= M时，表示边e顶点i开始的一部分已经在M距离内部。M-dist[i]是新增的节点。

当dist[j] <= M时，表示边e顶点j开始的一部分已经在M距离内部。M-dist[j]是新增的节点。

但是注意到，图中，从节点0到节点j的距离，不一定经过节点i。M-dist[i]与M-dist[j]个新增节点中，可能有重复。

要考虑到最大新增加节点数为距离权重w。

因此，新增节点数min(M-dist[i]+M-dist[j],w)。

构造图

用数组E表示边数组。

同样是用邻接表G表示图。但是链表中放置的是边索引。

G<节点i,[边1，边2,...]>

E=[];
G[i]=[];
k=0;
for(e in edges){
    <e[0],e[1],e[2]+1>入E[k];
    k入G[e[0]];
    k入G[e[1]];
    ++k;
}

求最短距离

求从节点0到其它节点的距离。

用dijkstra最短路径算法。用优先级队列Q挑选最短距离。

vis[i]=INT_MAX;
vis[0]=0;
Q=[];
<0,0>入Q;
while(Q不空){
    <x,x_dist>=Q.top弹出;
    for(j in G[x]){
        v=E[j]的另一个节点;
        if(vis[x]+E[j]的距离权重 < vis[v]){
            vis[v] = vis[x]+E[j]的距离权重;
            <v,vis[v]>入Q;
        }
    }
}

求最短距离<=M的节点个数

上面已经分析过。对边e(i,j)处理，如果到其顶点的最短距离 <= M，说明新增的节点要算入内。当然，原图中的节点也要算入内。

ans=0;结果
for(e in edges){
    a=0;
    if(vis[e[0]] <= M) a += M-vis[e[0]];
    if(vis[e[1]] <= M) a += M-vis[e[1]];
    ans = min(a,e[2]);
}
for(i < N){
    if(vis[i] <= M) ans++;
}

代码：

    struct node{
        int p;
        int d;
        node(int _p = 0,int _d=0):p(_p),d(_d){}
        bool operator<(const node& o)const{
            return d > o.d;
        }
    };
    struct edge{
        int u,v,w;
        edge(int _u=0,int _v=0,int _w=0):u(_u),v(_v),w(_w){}
        int another(int x)const{
            return x==u? v:u;
        }
    };
    typedef pair<int,int> Pair;
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        if(M ==0 ) return 1;
        vector<vector<int>> grid(N,vector<int>());
        vector<edge> _edges;
        for(int i=0;i<edges.size();++i){
            add_edge(grid,_edges,edges[i]);
        }
        
        vector<int> vis(N,INT_MAX);
        vis[0] = 0;
        priority_queue<node> Q;
        Q.push(node(0,0));
        while(Q.size()){
            node nd=Q.top();Q.pop();
            int p=nd.p;
            int d=nd.d;
            for(int i=0;i<grid[p].size();++i){
                edge e=_edges[grid[p][i]];
                int v=e.another(p);
                if(vis[p]+e.w < vis[v]){
                    vis[v] = vis[p]+e.w;
                    Q.push(node(v,vis[v]));
                }
            }
        }
        int ans=0;
        for(int i=0;i<_edges.size();++i){
            const edge& e=_edges[i];
            ans += min(max(M-vis[e.u],0)+max(M-vis[e.v],0),e.w-1);
        }
        for(int i=0;i<N;++i){
            if(vis[i] <= M) ++ans;
        }
        return ans;
    }
    
    void add_edge(vector<vector<int>>& grid,vector<edge>& _edges,vector<int>& v){
        edge e(v[0],v[1],v[2]+1);
        _edges.push_back(e);
        int i=_edges.size()-1;
        grid[e.u].push_back(i);
        grid[e.v].push_back(i);
    }
