解法一 回溯法（超时）

本题实际上是求，在每个位置i，放置A，L，P三个字符，并且满足限制条件。

这是全排列回溯算法的变形题。解法类似，全排列 和 全排列II。

在位置i放置字符，分情况讨论。

当位置i之前没有A时，位置i可放置A；

当位置i之前的两个位置不是LL时，位置i可放置L；

位置i可以放置P。

在每个位置考虑这三种情况，直到放置完成。

dfs(path,cur){
    if(cur == n) 一种合法排列，计数;
    else for(i in [A,L,P]){
        if(i== A){
            if(A in path[0,cur-1]) pass掉；
        }
        if(i==L 且 cur >= 2 且path[cur-2,cur-1] == LL) pass掉;
        path[cur]=i;
        dfs(path,cur+1);
    }
}

代码：

const int mode=1000000007;
class Solution {
public:
    int ans;
    int checkRecord(int n) {
        ans=0;
        vector<int> path(n+1,0);
        dfs(n,path,0);
        return ans;
    }
    
    void dfs(int n,vector<int>& path,int cur){
        if(cur == n){
            ans = (ans + 1)%mode;
        }else for(int i=1;i<=3;++i){
            bool ok=true;
            int c=0;
            if(i == 1){
                for(int j=0;j<cur;++j){
                    if(path[j] == 1){
                        ok=false;
                        break;
                    }
                }    
            }
            if(i == 2 && cur >= 2 && path[cur-1] == 2 && path[cur-2] == 2){
                ok=false;
            }
            if(ok){
                path[cur] = i;
                dfs(n,path,cur+1);
            }
        }
    }
};

解法二 回溯法

解法一会引入一些重复计算。要去除这些重复计算。

解法一在位置i，需要更多的信息，才能挖掘到重复计算。

在位置i，放置字符时，要明确两个条件。

1，位置i之前，有几个A；用变量a表示，取值范围[0,1].

2，位置i之前的两个位置i-2，i-1上，有几个L。 用变量b表示，取值范围[0,1,2].

对位置i的放置情况，分情况讨论

当a >0 时，不能放置A。

当a == 0 时，可以放置A。

放置A后，a=1。此时，不论之前b是多少，现在都变成b=0。

因此，在位置i+1，a=1，b=0.

当b==2时，不能放L。

当b < 2时，可以放置L。放置L，不会影响A。因此，a值不变，b=b+1.

因此，在位置i+1，a，b+1.

不轮a，b是多少，一定能放置P。

放置P，不影响A，a不变。不论之前b是多少，现在都变成0.

因此，在位置i+1，a，b=0.

dfs(cur,a,b){
    ans = 0;
    if(cur == n) 返回1；
    else for(i in [A,L,P]){
        if(i == A){
            if(a > 0 ) pass;
            ans += dfs(cur+1,1,0);
        }else if(i == L){
            if(b == 2 ) pass;
            ans += dfs(cur+1,a,b+1);
        }else{
            ans += dfs(cur+1,a,0);
        }
    }
    返回a;
}

代码（超时）：

const int mode=1000000007;
class Solution {
public:
    int ans;
    int checkRecord(int n) {
        ans=0;
        ans = dfs(n,0,0,0);
        return ans;
    }
    
    int dfs(int n,int cur,int a,int b){
        int ans = 0;
        if(cur == n){
            return 1;
        }else for(int i=1;i<=3;++i){
            if(i == 1) {
                if(a) continue;
                int r = dfs(n,cur+1,1,0);
                ans = (ans + r) % mode;
            }else if(i == 2) {
                if(b == 2) continue;
                int r = dfs(n,cur+1,a,b+1);
                ans = (ans + r) % mode;
            }else{
                int r = dfs(n,cur+1,a,0);
                ans = (ans + r) % mode;
            }
        }
        return ans;
    }
};

记忆化存储

在位置i，如果a和b值相同，算过一次后，不用重新计算。

设d[i][a][b]表示在位置i时，位置i之前有a个A，位置i之前相邻的两个位置有b个L，在这些情况下的排列数。

初始，i==n时，d[n][a][b]=1。其它d[i][a][b]=-1。

在上面的算法中，加上d，记忆化存储。

dfs(cur,a,b){
    ans = d[cur][a][b];
    if(ans > -1) 返回ans;
    ans = 0;
    if(cur == n) 返回1；
    else for(i in [A,L,P]){
        if(i == A){
            if(a > 0 ) pass;
            ans += dfs(cur+1,1,0);
        }else if(i == L){
            if(b == 2 ) pass;
            ans += dfs(cur+1,a,b+1);
        }else{
            ans += dfs(cur+1,a,0);
        }
    }
    返回a;
}

代码：

const int mode=1000000007;
class Solution {
public:
    int ans;
    int checkRecord(int n) {
        ans=0;
        vector<vector<vector<int>>> d(n+1,vector<vector<int>>(2,vector<int>(3,-1)));
        ans = dfs(n,0,0,0,d);
        
        return ans;
    }
    
    int dfs(int n,int cur,int a,int b,
            vector<vector<vector<int>>>& d){
        int& ans = d[cur][a][b];
        if(ans > -1) return ans;
        ans = 0;
        if(cur == n){
            return ans = 1;
        }else for(int i=1;i<=3;++i){
            if(i == 1) {
                if(a) continue;
                int r = dfs(n,cur+1,1,0,d);
                ans = (ans + r) % mode;
            }else if(i == 2) {
                if(b == 2) continue;
                int r = dfs(n,cur+1,a,b+1,d);
                ans = (ans + r) % mode;
            }else{
                int r = dfs(n,cur+1,a,0,d);
                ans = (ans + r) % mode;
            }
        }
        return ans;
    }
};

解法三 动态规划

将解法二，记忆化存储算法，改为递推。

从后往前递推

设d[i][a][b]表示在位置i时，位置i之前有a个A，位置i之前相邻的两个位置有b个L，在这些情况下的排列数。

初始，i==n时，d[n][a][b]=1。其它d[i][a][b]=0。

对位置i的放置情况，分情况讨论

当a >0 时，不能放置A。

当a == 0 时，可以放置A。

放置A后，a=1。此时，不论之前b是多少，现在都变成b=0。

因此，在位置i+1，a=1，b=0.

d[i][a][b] += d[i+1][1][0].

当b==2时，不能放L。

当b < 2时，可以放置L。放置L，不会影响A。因此，a值不变，b=b+1.

因此，在位置i+1，a，b+1.

d[i][a][b] += d[i+1][a][b+1].

不轮a，b是多少，一定能放置P。

放置P，不影响A，a不变。不论之前b是多少，现在都变成0.

因此，在位置i+1，a，b=0.

d[i][a][b] += d[i+1][a][0].

d[n][a][b]=1;
for(i in [n-1,...,0]){
    for(j in [0,1]){
        for(k in [0,1,2]){
            for(l in [A,L,P]){
                if(l == A){
                    if(j == 1) pass;
                    d[i][j][k] += d[i+1][1][0];
                }else if(l == L){
                    if(k == 2) pass;
                    d[i][j][k] += d[i+1][j][k+1];
                }else{
                    d[i][j][k] += d[i+1][j][0];
                }
            }
        }
    }
}
d[0][0][0]是结果;

代码：

const int mode=1000000007;
class Solution {
public:
    int checkRecord(int n) {
        vector<vector<vector<int>>> d(n+1,vector<vector<int>>(2,vector<int>(3,0)));
        for(int i=0;i<2;++i){
            for(int j=0;j<3;++j){
                d[n][i][j]=1;
            }
        }
        for(int i=n-1; i>=0; --i){
            for(int j=0;j<2;++j){
                for(int k=0;k<3;++k){
                    for(int l=1;l<=3;++l){
                        if(l==1){
                            if(j==1) continue;
                            int& ans=d[i][j][k];
                            ans = (ans + d[i+1][1][0]) % mode;
                        }else if(l==2){
                            if(k==2) continue;
                            int& ans=d[i][j][k];
                            ans = (ans + d[i+1][j][k+1]) % mode;
                        }else{
                            int& ans=d[i][j][k];
                            ans = (ans + d[i+1][j][0]) % mode;
                        }
                    }
                }                
            }
        }
        return d[0][0][0];
    }
};

从前往后递推

从上面的从后往前递推关系中，可以总结出下面的递推关系。

d[i][0][y]   <- d[i+1][1][0] (y=0,1,2)
d[i][x][0,1] <- d[i+1][x][1,2] (x=0,1)
d[i][x][y]   <- d[i+1][x][0] (x=0,1,y=0,1,2)

因此，改为从前到后，可总结出下面的递推关系。

d[i-1][0][y] -> d[i][1][0] (y=0,1,2)
d[i-1][x][0] -> d[i][x][1] (x=0,1)
d[i-1][x][1] -> d[i][x][2] (x=0,1)
d[i-1][x][y] -> d[i][x][0] (x=0,1,y=0,1,2)

对箭头右边，将x=0，1展开，并整理得

d[i-1][0][y] -> d[i][0][0] (y=0,1,2)
d[i-1][0][0] -> d[i][0][1]
d[i-1][0][1] -> d[i][0][2]
d[i-1][0][y] -> d[i][1][0] (y=0,1,2)
d[i-1][1][y] -> d[i][1][0] (y=0,1,2)
d[i-1][1][0] -> d[i][1][1]
d[i-1][1][1] -> d[i][1][2]

容易得出递推思路：

d[0][0][0]=1;
d[0][0][1] = 1;
d[0][1][0] = 1; 

for(i in [1,n-1]){
    d[i][0][0] = d[i-1][0][0] + d[i-1][0][1] + d[i-1][0][2];
    d[i][0][1] = d[i-1][0][0];
    d[i][0][2] = d[i-1][0][1];
    d[i][1][0] = d[i-1][0][0] + d[i-1][0][1] + d[i-1][0][2] + d[i-1][1][0] + d[i-1][1][1] + d[i-1][1][2];
    d[i][1][1] = d[i-1][1][0];
    d[i][1][2] = d[i-1][1][1];
}
结果是d[n-1][x][y]的和。

代码：

    int checkRecord(int n) {
        const int mode=1000000007;
        vector<vector<vector<int>>> d(n+1,vector<vector<int>>(2,vector<int>(3,0)));
        d[0][0][0]=1;
        d[0][0][1] = 1;
        d[0][1][0] = 1; 
        for(int i=1; i <= n; ++i){
            for(int j=0;j<3;++j){
                d[i][0][0] = (d[i][0][0] + d[i-1][0][j]%mode)%mode;
            }
            
            d[i][0][1] = d[i-1][0][0];
            d[i][0][2] = d[i-1][0][1];
            for(int j=0;j<2;++j){
                for(int k=0;k<3;++k){
                    d[i][1][0] = (d[i][1][0] + d[i-1][j][k]%mode)%mode;
                }                
            }
            d[i][1][1] = d[i-1][1][0];
            d[i][1][2] = d[i-1][1][1];
        }

        return d[n][1][0];
    }

降到二维滚动数组

从上面可看出，d[i]只与d[i-1]有关。可直接将d的第一个维度改为2.

进一步，直接将d将为二维也是可以的。

d[0][0]=1;
d[0][1] = 1;
d[1][0] = 1; 

for(i in [1,n-1]){
    a = d[0][0] + d[0][1] + d[0][2];
    b = d[0][0];
    c = d[0][1];
    d = d[0][0] + d[0][1] + d[0][2] + d[1][0] + d[1][1] + d[1][2];
    e = d[1][0];
    f = d[1][1];
    d[0][0] = a;
    d[0][1] = b;
    d[0][2] = c;
    d[1][0] = d;
    d[1][1] = e;
    d[1][2] = f;
}
结果是d[x][y]的和。

代码：

    int checkRecord(int n) {
        const int mode=1000000007;
        vector<vector<int>> d(2,vector<int>(3,0));
        d[0][0]=1;
        d[0][1] = 1;
        d[1][0] = 1; 
        for(int i=1; i <= n; ++i){
            int a=0;
            for(int j=0;j<3;++j){
                a = (a + d[0][j]%mode)%mode;
            }
            
            int b = d[0][0];
            int c = d[0][1];
            int dx = 0;
            for(int j=0;j<2;++j){
                for(int k=0;k<3;++k){
                    dx = (dx + d[j][k]%mode)%mode;
                }                
            }
            int e = d[1][0];
            int f = d[1][1];
            d[0][0] = a;
            d[0][1] = b;
            d[0][2] = c;
            d[1][0] = dx;
            d[1][1] = e;
            d[1][2] = f;            
        }

        return d[1][0];
    }
