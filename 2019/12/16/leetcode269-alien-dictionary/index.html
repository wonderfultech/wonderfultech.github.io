解法一 拓扑排序

从已经有序的字典中，推断出字符顺序。

关键是明确字符串比较方法。

字符串A和B比较大小。

在相同位置i处，

当A[i] < B[i]时，A小于B。

因此，反过来，A和B在相同位置i上，只有A[i] != B[i]时，才能有字符串A和B的顺序推断出字符A[i]和B[i]的大小关系。

思路：对字典中的每个字符串A，与其之后的每个字符串B，组合。找到A和B中，第一对，在相同位置i上，A[i] ！= B[i]的两个字符串。确定A[i] < B[i]。并以此构造一个有向边A[i]->B[i]。

由这些有向边，构造了一个有向图G。

此外，不在任何有向边中，但是在字典中的其它字符。它们的顺序并不重要，也可以将它们加入图G。

最后对G拓扑排序。就能明确字符顺序。

构造图

字典中的字符都是小写字母，最多256个字符。

用邻接矩阵就能表示有向图。

G[256][256]=0;//G[u][v]=1，表示有向边u->v
has[256]=0;表示G中是否有字符a;

for(A in words){
    for(B in words中A之后的字符串){
        len=min(A.length,B.length);
        k=0;
        while(k < len){
            ++has[A[k]];
            ++has[B[k]];
            if(A[k] != B[k]){
                G[A[k]][B[k]]=1;
                break;
            }
            ++k;
        }
        //将不在有向边中的字符也加入图
        nk = k;
        while(k < A.length) ++has[A[k++]];
        while(nk < B.length) ++has[B[nk++]];
    }
}

拓扑排序算法

用递归方法求拓扑排序。同时，排除环的情况。

按有向边，遍历图中节点u的全部邻接点v。

//vis[i]=0，未遍历过
//vis[i]=-1,正在遍历
//vis[i]=1,遍历完成
vis[i]=0;
path=[];//拓扑排序，节点放置位置
cur=256;//拓扑排序，节点位置cur
dfs(G,has,vis,path,cur,u){
    vis[u]=-1;
    for(v < 256){
        if(v not in has 或 G[u][v]==0) pass掉;
        if(vis[v] < 0) 有环，返回失败;
        else if(vis[v] == 0 且 dfs(G,has,vis,path,cur,v)失败) 递归遍历v失败，返回失败;
    }
    path[--cur]=u;
    vis[u]=1;
    返回成功。
}

生成结果

对图中节点，进行拓扑排序。

cur=256;
for(u < 256){
    if(has[u] == 1 且 vis[u] == 0){
        rt=dfs(G,has,vis,path,cur);
        if(rt==false)有环，返回"";
    }
}
path[cur:255]就是结果.

代码：

    string alienOrder(vector<string>& words) {
        if(words.size()==1){
            sort(words[0].begin(),words[0].end());
            string ans;
            for(int i=0;i<words[0].size();++i){
                if(!i || words[0][i] != words[0][i-1]){
                    ans += words[0][i];
                }
            }
            return ans;
        }
        vector<vector<int>> grid(256,vector<int>(256,0));
        vector<int> has(256,0);
        for(int i=0;i<words.size();++i){
            for(int j=i+1;j<words.size();++j){
                int k=0;
                const int len=min(words[i].size(),words[j].size());
                while(k < len){
                    int a=words[i][k];
                    int b=words[j][k];
                    ++has[a];
                    ++has[b];
                    if(a != b){
                        grid[a][b]=1;
                        break;
                    }
                    ++k;
                }
                int nk=k;
                while(k < words[i].size()) ++has[words[i][k++]];
                while(nk < words[j].size()) ++has[words[j][nk++]];
            }
        }
       
        vector<int> vis(256,0);
        vector<int> path(256);
        int cur=256;
        for(int i=0;i<256;++i){
            if(has[i] && !vis[i]){
                bool rt=dfs(grid,has,vis,path,cur,i);
                if(!rt) return "";
            }
        }
        string ans;
        while(cur < 256){
            ans += path[cur++];
        }
        return ans;
    }
    
    bool dfs(vector<vector<int>>& grid,
             vector<int>& has,
             vector<int>& vis,
             vector<int>& path,
             int& cur,
             int u){
        vis[u]=-1;
        for(int v=0;v<grid[u].size();++v){
            if(!has[v] || !grid[u][v]) continue;
            if(vis[v] < 0) return false;
            else if(!vis[v] && !dfs(grid,has,vis,path,cur,v)) return false;
        }
        path[--cur]=u;
        vis[u]=1;
        return true;
    }
