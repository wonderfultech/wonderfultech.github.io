leetcode208 实现 Trie (前缀树) Implement Trie (Prefix Tree)

Posted on 2019年11月3日2019年12月4日by jasonpeng

leetcode 数据库题目全部题解

解法一 多叉树法

Trie树是多叉树的一种.

Trie树的重要特点:

1, 父节点到子节点的边表示字母. 而不是子节点存储字母. 这一点不好理解.

2, 一个完整字符串的叶节点要有标记. 标记表示有从根节点到此的字符串.

节点结构:

26个指针+标记

    struct node{
        node* next[26];
        int t;
        node(int _t=0):t(_t){
            memset(next,0,sizeof(next));
        }
    };

插入字符串:

node root;根节点;
insert(s){
    p=root;
    for(ch in s){
        if(p.next[ch]为空){//插入字母ch
            p.next[ch] = node();
        }
        p = p.next[ch];
    }
    //新插入串
    p.t = 1;
}

搜索字符串:

node root;根节点;
search(s){
    p=root;
    for(ch in s){
        if(p.next[ch]为空){//插入字母ch
            返回失败;
        }
        p = p.next[ch];
    }
    //是否完整串
    返回p.t == 1;
}

搜索前缀:

node root;根节点;
search(s){
    p=root;
    for(ch in s){
        if(p.next[ch]为空){//插入字母ch
            返回失败;
        }
        p = p.next[ch];
    }
    返回 p是否为空;
}

代码:

    struct node{
        node* next[26];
        int t;
        node(int _t=0):t(_t){
            memset(next,0,sizeof(next));
        }
    };
    node root;
    /** Initialize your data structure here. */
    Trie() :root(){
        
    }
    
    int get_id(char ch){
        return ch-'a';
    }
    
    /** Inserts a word into the trie. */
    void insert(string word) {
        node* p=&root;
        for(int i=0;i<word.size();++i){
            int id=get_id(word[i]);
            if(!p->next[id]){
                p->next[id] = new node();
            }
            p = p->next[id];
        }
        p->t = 1;
    }
    
    /** Returns if the word is in the trie. */
    bool search(string word) {
        node* p=&root;
        for(int i=0;i<word.size();++i){
            int id=get_id(word[i]);
            if(!p->next[id]){
                return false;
            }
            p = p->next[id];
        }
        return p ? p->t == 1: false;
    }
    
    /** Returns if there is any word in the trie that starts with the given prefix. */
    bool startsWith(string prefix) {
        node* p=&root;
        for(int i=0;i<prefix.size();++i){
            int id=get_id(prefix[i]);
            if(!p->next[id]){
                return false;
            }
            p = p->next[id];
        }
        return p != NULL;
    }
