leetcode1162 地图分析 As Far from Land as Possible

Posted on 2019年11月12日2019年12月4日by jasonpeng

leetcode 数据库题目全部题解

解法一 广度优先搜索

无向图上的最短路径问题。

从每个陆地S出发。计算能与S连通的海洋离S的距离D。

这样每个海洋有多个距离值[D1,D2,...,DX]。

求这些距离中的最小值。

完整思路：

vis[x][y] = 0;去重
h[x][y] = 0;海洋的距离值
Q;队列
bfs(grid,x,y,vis,h){
    (x,y)入Q;
    while(Q不空){
        (a,b)=Q.head弹出;
        for((c,d) in (a,b)的四周){
            if((c,d)在图中 且 grid(c,d)是海洋 且 vis(c,d)没遍历过){
                if(h(c,d) ==0 或 h(a,b)+1 < h(c,d)){
                    h(c,d) = h(a,b)+1;
                    (c,d)入Q;
                }
            }
        }
    }
}

代码：

    int R,C; 
    int maxDistance(vector<vector<int>>& grid) {
        if(!(R=grid.size()) || !(C=grid[0].size())) return 0;
        vector<vector<int>> vis(R,vector<int>(C,0));
        vector<vector<int>> h(R,vector<int>(C,0));
        int c=0;
        for(int i=0;i<R;++i){
            for(int j=0;j<C;++j){
                if(grid[i][j]){
                    bfs(grid,vis,i,j,++c,h);
                }
            }
        }
        int ans = 0;
        for(int i=0;i<R;++i){
            for(int j=0;j<C;++j){
                if(grid[i][j] == 0){
                    ans = max(ans,h[i][j]);
                }
            }
        }
        return ans ? ans : -1;
    }
    
    typedef pair<int,int> Pair;
    void bfs(const vector<vector<int>>& grid,
           vector<vector<int>>& vis,
             int x,int y,int c,
           vector<vector<int>>& h){
        const static int dx[4]={1,-1,0,0};
        const static int dy[4]={0,0,1,-1};
        queue<Pair> Q;
        Q.push(make_pair(x,y));
        while(Q.size()){
            Pair cur = Q.front();Q.pop();
            int& _x = cur.first;
            int& _y = cur.second;
            for(int i=0;i<4;++i){
                const int nx = _x+ dx[i];
                const int ny = _y+ dy[i];
                if(in_grid(nx,ny) && grid[nx][ny] == 0 && vis[nx][ny] < c){
                    if(h[nx][ny] == 0 || h[_x][_y]+1 < h[nx][ny]){
                        vis[nx][ny] = c;
                        h[nx][ny] = h[_x][_y]+1;
                        Q.push(make_pair(nx,ny));
                    }
                }
            }
        }        
    }
    
    bool in_grid(int x,int y){
        return x>=0 && x<R && y>=0 && y<C;
    }
